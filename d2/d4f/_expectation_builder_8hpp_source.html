181 {

184 "For non-void return types, a finalize-policy must be specified. "
185 "See: https://dnkpp.github.io/mimicpp/db/d7a/group___e_x_p_e_c_t_a_t_i_o_n___f_i_n_a_l_i_z_e_r.html#details");

187 return ScopedExpectation{
188 std::move(m_Storage),
189 std::apply(
190 [&](auto&... policies) {
191 ControlPolicy controlPolicy{
192 std::move(m_TimesConfig),
193 std::move(m_SequenceConfig)};
194
195 using ExpectationT = BasicExpectation<
196 Signature,
197 decltype(controlPolicy),
198 FinalizePolicy,
199 Policies...>;
200
201 return std::make_unique<ExpectationT>(
202 std::move(sourceLocation),
203 std::move(m_TargetReport),
204 std::move(controlPolicy),
205 std::move(m_FinalizePolicy),
206 std::move(policies)...);
207 },
208 m_ExpectationPolicies)};
209 }

223 template <typename Param, typename... Canary, matcher_for<Param> Arg>
224 [[nodiscard]]
225 constexpr auto make_arg_matcher([[maybe_unused]] util::priority_tag<2> const, Arg arg)
226 {
227 return arg;
228 }
229
230 // if the param is a character-pointer, there is no evidence, whether it denotes a null-terminated string or just an
231 // actual pointer to a value.
232 // But, the Mock user shall know it, thus if `Arg` is not a character-pointer, we enable this matcher.
233 template <string Param, typename... Canary, string Arg>
234 requires(!std::is_pointer_v<std::remove_reference_t<Param>>)
235 || (!std::is_pointer_v<std::remove_reference_t<Arg>>)
236 [[nodiscard]] //
237 constexpr auto make_arg_matcher([[maybe_unused]] util::priority_tag<1> const, Arg&& arg)
238 {
239 return matches::str::eq(std::forward<Arg>(arg));
240 }
241
242 template <typename Param, typename... Canary, util::weakly_equality_comparable_with<Param> Arg>
243 [[nodiscard]]
244 constexpr auto make_arg_matcher([[maybe_unused]] util::priority_tag<0> const, Arg&& arg)
245 {
246 return matches::eq(std::forward<Arg>(arg));
247 }
248
249 inline constexpr util::priority_tag<2> maxMakeArgMatcherTag{};
250
251 template <typename Arg, typename Target>
252 concept requirement_for = requires {
253 {
254 detail::make_arg_matcher<Target>(maxMakeArgMatcherTag, std::declval<Arg>())
255 } -> matcher_for<Target>;
256 };
257
258 template <
259 typename Signature,
260 std::size_t index,
261 typename Arg,
262 typename... Canary,
263 typename Param = signature_param_type_t<index, Signature>>
264 requires requirement_for<Arg, Param>
265 constexpr auto make_arg_policy(Arg&& arg)
266 {
267 return expect::arg<index>(
268 detail::make_arg_matcher<Param>(maxMakeArgMatcherTag, std::forward<Arg>(arg)));
269 }
270
271 template <typename Signature, typename Builder, std::size_t... indices, typename... Args>
272 [[nodiscard]]
273 constexpr auto extend_builder_with_arg_policies(
274 Builder&& builder,
275 [[maybe_unused]] std::index_sequence<indices...> const,
276 Args&&... args)
277 {
278 return (
279 std::forward<Builder>(builder)
280 && ...
281 && detail::make_arg_policy<Signature, indices>(std::forward<Args>(args)));
282 }

301 sequence::detail::Config<>{},
302 expectation_policies::InitFinalize{},
303 std::tuple{}},
304 std::index_sequence_for<Args...>{},
305 std::forward<Args>(args)...);
