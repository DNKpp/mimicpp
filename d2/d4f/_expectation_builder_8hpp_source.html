28 template <
29 bool timesConfigured,
30 typename SequenceConfig,
31 typename Signature,
32 typename FinalizePolicy,
33 expectation_policy_for<Signature>... Policies>
34 class BasicExpectationBuilder
35 {
36 public:
37 using StorageT = ExpectationCollection<Signature>;
38 using PolicyListT = std::tuple<Policies...>;
39 using ReturnT = typename Expectation<Signature>::ReturnT;
40
41 ~BasicExpectationBuilder() = default;

43 template <typename FinalizePolicyArg, typename PolicyListArg>
44 requires std::constructible_from<FinalizePolicy, FinalizePolicyArg>
45 && std::constructible_from<PolicyListT, PolicyListArg>
46 [[nodiscard]]
47 explicit constexpr BasicExpectationBuilder(
48 std::shared_ptr<StorageT> storage,
49 reporting::TargetReport target,
50 detail::TimesConfig timesConfig,
51 SequenceConfig sequenceConfig,
52 FinalizePolicyArg&& finalizePolicyArg,
53 PolicyListArg&& policyListArg) noexcept
54 : m_Storage{std::move(storage)},
55 m_TargetReport{std::move(target)},
56 m_TimesConfig{std::move(timesConfig)},
57 m_SequenceConfig{std::move(sequenceConfig)},
58 m_FinalizePolicy{std::forward<FinalizePolicyArg>(finalizePolicyArg)},
59 m_ExpectationPolicies{std::forward<PolicyListArg>(policyListArg)}
60 {
61 MIMICPP_ASSERT(m_Storage, "Storage is nullptr.");

71 template <typename Policy>
72 requires std::same_as<expectation_policies::InitFinalize, FinalizePolicy>
73 && (!std::same_as<expectation_policies::InitFinalize, std::remove_cvref_t<Policy>>)
74 && finalize_policy_for<std::remove_cvref_t<Policy>, Signature>
75 [[nodiscard]]
76 friend constexpr auto operator&&(BasicExpectationBuilder&& builder, Policy&& policy)
77 {
78 using ExtendedExpectationBuilderT = BasicExpectationBuilder<
79 timesConfigured,
80 SequenceConfig,
81 Signature,
82 std::remove_cvref_t<Policy>,
83 Policies...>;
84
85 return ExtendedExpectationBuilderT{
86 std::move(builder.m_Storage),
87 std::move(builder.m_TargetReport),
88 std::move(builder.m_TimesConfig),
89 std::move(builder.m_SequenceConfig),
90 std::forward<Policy>(policy),
91 std::move(builder.m_ExpectationPolicies)};
92 }

94 template <typename Policy>
95 requires expectation_policy_for<std::remove_cvref_t<Policy>, Signature>
96 [[nodiscard]]
97 friend constexpr auto operator&&(BasicExpectationBuilder&& builder, Policy&& policy)
98 {
99 using ExtendedExpectationBuilderT = BasicExpectationBuilder<
100 timesConfigured,
101 SequenceConfig,
102 Signature,
103 FinalizePolicy,
104 Policies...,
105 std::remove_cvref_t<Policy>>;
106
107 return ExtendedExpectationBuilderT{
108 std::move(builder.m_Storage),
109 std::move(builder.m_TargetReport),
110 std::move(builder.m_TimesConfig),
111 std::move(builder.m_SequenceConfig),
112 std::move(builder.m_FinalizePolicy),
113 std::tuple_cat(
114 std::move(builder.m_ExpectationPolicies),
115 std::forward_as_tuple(std::forward<Policy>(policy)))};
116 }

136 }

140 friend constexpr auto operator&&(BasicExpectationBuilder&& builder, sequence::detail::Config<Sequences...>&& config)
141 {
142 sequence::detail::Config newConfig = builder.m_SequenceConfig.concat(std::move(config));
143
144 using ExtendedExpectationBuilderT = BasicExpectationBuilder<
145 timesConfigured,
146 decltype(newConfig),
147 Signature,
148 FinalizePolicy,
149 Policies...>;
150
151 return ExtendedExpectationBuilderT{
152 std::move(builder.m_Storage),
153 std::move(builder.m_TargetReport),
154 std::move(builder.m_TimesConfig),
155 std::move(newConfig),
156 std::move(builder.m_FinalizePolicy),
157 std::move(builder.m_ExpectationPolicies)};
158 }
159
160 [[nodiscard]]
161 ScopedExpectation finalize(util::SourceLocation sourceLocation) &&
162 {
163 static_assert(
164 finalize_policy_for<FinalizePolicy, Signature>,
165 "For non-void return types, a finalize policy must be set.");
166
167 return ScopedExpectation{
168 std::move(m_Storage),
169 std::apply(
170 [&](auto&... policies) {
171 ControlPolicy controlPolicy{
172 std::move(m_TimesConfig),
173 std::move(m_SequenceConfig)};
174
175 using ExpectationT = BasicExpectation<
176 Signature,
177 decltype(controlPolicy),
178 FinalizePolicy,
179 Policies...>;
180
181 return std::make_unique<ExpectationT>(
182 std::move(sourceLocation),
183 std::move(m_TargetReport),
184 std::move(controlPolicy),
185 std::move(m_FinalizePolicy),
186 std::move(policies)...);
187 },
188 m_ExpectationPolicies)};
189 }
190
191 private:
192 std::shared_ptr<StorageT> m_Storage;
193 reporting::TargetReport m_TargetReport;
194 detail::TimesConfig m_TimesConfig{};
195 SequenceConfig m_SequenceConfig{};
196 FinalizePolicy m_FinalizePolicy{};
197 PolicyListT m_ExpectationPolicies{};
198 };

203 template <typename Param, typename Arg>
204 requires matcher_for<
205 std::remove_cvref_t<Arg>,
206 Param>
207 [[nodiscard]]
208 constexpr auto make_arg_matcher([[maybe_unused]] util::priority_tag<2> const, Arg&& arg)
209 {
210 return std::forward<Arg>(arg);
211 }
212
213 // if the param is a character-pointer, there is no evidence, whether it denotes a null-terminated string or just an
214 // actual pointer to a value.
215 // But, the Mock user shall know it, thus if `Arg` is not a character-pointer, we enable this matcher.
216 template <string Param, string Arg>
217 requires(!std::is_pointer_v<std::remove_reference_t<Param>>)
218 || (!std::is_pointer_v<std::remove_reference_t<Arg>>)
219 [[nodiscard]] constexpr auto make_arg_matcher([[maybe_unused]] util::priority_tag<1> const, Arg&& arg)
220 {
221 return matches::str::eq(std::forward<Arg>(arg));
222 }
223
224 template <typename Param, util::weakly_equality_comparable_with<Param> Arg>
225 [[nodiscard]]
226 constexpr auto make_arg_matcher([[maybe_unused]] util::priority_tag<0> const, Arg&& arg)
227 {
228 return matches::eq(std::forward<Arg>(arg));
229 }
230
231 constexpr util::priority_tag<2> max_make_arg_matcher_tag{};
232
233 template <typename Arg, typename Target>
234 concept requirement_for = requires {
235 {
236 detail::make_arg_matcher<Target, Arg>(max_make_arg_matcher_tag, std::declval<Arg>())
237 } -> matcher_for<Target>;
238 };
239
240 template <
241 typename Signature,
242 std::size_t index,
243 typename Arg,
244 typename... Canary,
245 typename Param = signature_param_type_t<index, Signature>>
246 requires requirement_for<Arg, Param>
247 constexpr auto make_arg_policy(Arg&& arg)
248 {
249 return expect::arg<index>(
250 detail::make_arg_matcher<Param, Arg>(max_make_arg_matcher_tag, std::forward<Arg>(arg)));
251 }
252
253 template <typename Signature, typename Builder, std::size_t... indices, typename... Args>
254 [[nodiscard]]
255 constexpr auto extend_builder_with_arg_policies(
256 Builder&& builder,
257 const std::index_sequence<indices...>,
258 Args&&... args)
259 {
260 return (
261 std::forward<Builder>(builder)
262 && ...
263 && detail::make_arg_policy<Signature, indices>(
264 std::forward<Args>(args)));
265 }
266
267 template <typename Signature, typename... Args>
268 constexpr auto make_expectation_builder(
269 std::shared_ptr<ExpectationCollection<Signature>> expectations,
270 reporting::TargetReport target,
271 Args&&... args)
272 {
273 using BaseBuilderT = BasicExpectationBuilder<
274 false,
275 sequence::detail::Config<>,
276 Signature,
277 expectation_policies::InitFinalize>;
278
279 return detail::extend_builder_with_arg_policies<Signature>(
280 BaseBuilderT{
281 std::move(expectations),
282 std::move(target),
283 TimesConfig{},
284 sequence::detail::Config<>{},
285 expectation_policies::InitFinalize{},
286 std::tuple{}},
287 std::index_sequence_for<Args...>{},
288 std::forward<Args>(args)...);
289 }
