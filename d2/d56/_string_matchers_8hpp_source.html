33 template <string Target, string Pattern>
34 constexpr void check_string_compatibility() noexcept
35 {
36 static_assert(
37 std::same_as<
38 string_char_t<Target>,
39 string_char_t<Pattern>>,
40 "Pattern and target string must have the same character-type.");
41 }

43 struct make_view_fn
44 {
45 template <string T>
46 [[nodiscard]]
47 constexpr auto operator()(T const& str) const
48 {
49 return string_traits<T>::view(str);
50 }
51 };

53 template <string String>

55 constexpr auto make_view(String const& str)

300 return PredicateMatcher{

302 detail::check_string_compatibility<T, Pattern>();
303
