66 class DefaultReporter final
67 : public IReporter
68 {
69 public:
70 [[nodiscard]]
71 DefaultReporter() = default;
72
73 [[nodiscard]]
74 explicit DefaultReporter(std::ostream& out) noexcept
75 : m_Out{std::addressof(out)}
76 {
77 }
78
79 [[noreturn]]
80 void report_no_matches(
81 CallReport call,
82 std::vector<NoMatchReport> noMatchReports) override
83 {
84 auto const msg = stringify_no_matches(call, noMatchReports);
85 if (m_Out)
86 {
87 *m_Out << msg << '\n';
88 }
89
90 std::source_location const loc{*call.fromLoc};
91 throw UnmatchedCallT{msg, std::move(call), loc};
92 }
93
94 [[noreturn]]
95 void report_inapplicable_matches(
96 CallReport call,
97 std::vector<ExpectationReport> expectationReports) override
98
99 {
100 const auto msg = stringify_inapplicable_matches(call, expectationReports);
101 if (m_Out)
102 {
103 *m_Out << msg << '\n';
104 }
105
106 std::source_location const loc{*call.fromLoc};
107 throw UnmatchedCallT{msg, std::move(call), loc};
108 }
109
110 void report_full_match(
111 [[maybe_unused]] CallReport const call,
112 [[maybe_unused]] ExpectationReport const expectationReport) noexcept override
113 {
114 assert(std::holds_alternative<state_applicable>(expectationReport.controlReport) && "Report denotes inapplicable expectation.");
115 }
116
117 void report_unfulfilled_expectation(ExpectationReport expectationReport) override
118 {
119 if (0 == std::uncaught_exceptions())
120 {
121 auto const msg = stringify_unfulfilled_expectation(expectationReport);
122 if (m_Out)
123 {
124 *m_Out << msg << '\n';
125 }
126
127 throw UnfulfilledExpectationT{msg, std::move(expectationReport)};
128 }
129 }
130
131 void report_error(const StringT message) override
132 {
133 if (0 == std::uncaught_exceptions())
134 {
135 if (m_Out)
136 {
137 *m_Out << message << '\n';
138 }
139
140 throw Error{message};
141 }
142 }
143
144 void report_unhandled_exception(
145 const CallReport call,
146 const ExpectationReport expectationReport,
147 const std::exception_ptr exception) override
148 {
149 if (m_Out)
150 {
151 *m_Out << stringify_unhandled_exception(call, expectationReport, exception)
152 << '\n';
153 }
154 }
155
156 private:
157 std::ostream* m_Out{};
158 };
159}
