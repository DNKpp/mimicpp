25 template <typename Matcher>
26 struct matcher_matches_fn
27 {
28 public:
29 Matcher const& matcher;
30
31 template <typename... Args>
32 requires std::invocable<
33 decltype(detail::matches_hook::matches),
34 const Matcher&,
35 Args&...>
36 [[nodiscard]]
37 // projected arguments may come as value, so Args& won't work in all cases
38 // just forward them as lvalue-ref
39 constexpr bool operator()(Args&&... args) const
40 noexcept(
41 std::is_nothrow_invocable_v<
42 decltype(detail::matches_hook::matches),
43 Matcher const&,
44 Args&...>)
45 {
46 return detail::matches_hook::matches(matcher, args...);
47 }
48 };

50 template <typename Matcher, typename MatchesStrategy, typename DescribeStrategy>
51 class ArgsRequirement
52 {
53 public:
54 [[nodiscard]]
55 explicit constexpr ArgsRequirement(
56 Matcher matcher,
57 MatchesStrategy matchesStrategy,
58 DescribeStrategy describeStrategy)
59 noexcept(
60 std::is_nothrow_move_constructible_v<Matcher>
61 && std::is_nothrow_move_constructible_v<MatchesStrategy>
62 && std::is_nothrow_move_constructible_v<DescribeStrategy>)
63 : m_Matcher{std::move(matcher)},
64 m_MatchesStrategy{std::move(matchesStrategy)},
65 m_DescribeStrategy{std::move(describeStrategy)}
66 {
67 }
68
69 [[nodiscard]]
70 static constexpr bool is_satisfied() noexcept
71 {
72 return true;
73 }
74
75 template <typename Return, typename... Args>
76 requires std::is_invocable_r_v<bool, MatchesStrategy const&, matcher_matches_fn<Matcher>, call::Info<Return, Args...> const&>
77 [[nodiscard]]
78 constexpr bool matches(const call::Info<Return, Args...>& info) const
79 noexcept(std::is_nothrow_invocable_v<MatchesStrategy const&, matcher_matches_fn<Matcher>, call::Info<Return, Args...> const&>)
80 {
81 return std::invoke(
82 m_MatchesStrategy,
83 matcher_matches_fn<Matcher>{m_Matcher},
84 info);
85 }
86
87 template <typename Return, typename... Args>
88 static constexpr void consume([[maybe_unused]] call::Info<Return, Args...> const& info) noexcept
89 {
90 }
91
92 [[nodiscard]]
93 std::optional<StringT> describe() const
94 {
95 [[maybe_unused]] auto const description = detail::describe_hook::describe(m_Matcher);
96
97 if constexpr (util::boolean_testable<decltype(description)>)
98 {
99 if (description)
100 {
101 return std::invoke(m_DescribeStrategy, *description);
102 }
103
104 return std::nullopt;
105 }
106 else if constexpr (std::convertible_to<decltype(description), StringViewT>)
107 {
108 return std::invoke(m_DescribeStrategy, description);
109 }
110 else
111 {
112 return std::nullopt;
113 }
114 }
115
116 private:
117 Matcher m_Matcher;
118 [[no_unique_address]] MatchesStrategy m_MatchesStrategy;
119 [[no_unique_address]] DescribeStrategy m_DescribeStrategy;
120 };

152 template <
153 std::size_t... indices,
154 typename Matcher,
155 typename... Projections>
156 [[nodiscard]]
157 constexpr auto make_args_policy(
158 Matcher&& matcher,
159 std::tuple<Projections...>&& projections)
160 {
161 static_assert(
162 sizeof...(indices) == sizeof...(Projections),
163 "Indices and projections size mismatch.");
164
165 using arg_selector_t = mimicpp::detail::args_selector_fn<
166 std::add_lvalue_reference_t,
167 std::index_sequence<indices...>>;
168 using apply_strategy_t = mimicpp::detail::arg_list_indirect_apply_fn<std::remove_cvref_t<Projections>...>;
169 using describe_strategy_t = arg_requirement_describer<indices...>;
170
171 return expectation_policies::ArgsRequirement{
172 std::forward<Matcher>(matcher),
173 mimicpp::detail::apply_args_fn(
174 arg_selector_t{},
175 apply_strategy_t{std::move(projections)}),
176 describe_strategy_t{}};
177 }
178 }

239 template <
240 std::size_t first,
241 std::size_t... others,
242 typename Matcher,
243 typename... Projections>
244 [[nodiscard]]
245 constexpr auto args(Matcher&& matcher, Projections&&... projections)
246 noexcept(
247 std::is_nothrow_constructible_v<std::remove_cvref_t<Matcher>, Matcher>
248 && (... && std::is_nothrow_move_constructible_v<std::remove_cvref_t<Projections>>))
249 {
250 static_assert(
251 sizeof...(projections) <= 1u + sizeof...(others),
252 "The projection count exceeds the amount of indices.");
253
254 return detail::make_args_policy<first, others...>(
255 std::forward<Matcher>(matcher),
256 util::detail::expand_tuple<std::identity, 1u + sizeof...(others)>(
257 std::forward_as_tuple(std::forward<Projections>(projections)...)));
258 }
