24 template <typename Matcher>
25 struct matcher_matches_fn
26 {
27 public:
28 const Matcher& matcher;
29
30 template <typename... Args>
31 requires std::invocable<
32 decltype(detail::matches_hook::matches),
33 const Matcher&,
34 Args&...>
35 [[nodiscard]]
36 // projected arguments may come as value, so Args& won't work in all cases
37 // just forward them as lvalue-ref
38 constexpr bool operator()(Args&&... args) const
39 noexcept(
40 std::is_nothrow_invocable_v<
41 decltype(detail::matches_hook::matches),
42 const Matcher&,
43 Args&...>)
44 {
45 return detail::matches_hook::matches(matcher, args...);
46 }
47 };

49 template <typename Matcher, typename MatchesStrategy, typename DescribeStrategy>
50 class ArgsRequirement
51 {
52 public:
53 [[nodiscard]]
54 explicit constexpr ArgsRequirement(
55 Matcher matcher,
56 MatchesStrategy matchesStrategy,
57 DescribeStrategy describeStrategy)
58 noexcept(
59 std::is_nothrow_move_constructible_v<Matcher>
60 && std::is_nothrow_move_constructible_v<MatchesStrategy>
61 && std::is_nothrow_move_constructible_v<DescribeStrategy>)
62 : m_Matcher{std::move(matcher)},
63 m_MatchesStrategy{std::move(matchesStrategy)},
64 m_DescribeStrategy{std::move(describeStrategy)}
65 {
66 }
67
68 [[nodiscard]]
69 static constexpr bool is_satisfied() noexcept
70 {
71 return true;
72 }
73
74 template <typename Return, typename... Args>
75 requires std::is_invocable_r_v<bool, const MatchesStrategy&, matcher_matches_fn<Matcher>, const call::Info<Return, Args...>&>
76 [[nodiscard]]
77 constexpr bool matches(const call::Info<Return, Args...>& info) const
78 noexcept(std::is_nothrow_invocable_v<const MatchesStrategy&, matcher_matches_fn<Matcher>, const call::Info<Return, Args...>&>)
79 {
80 return std::invoke(
81 m_MatchesStrategy,
82 matcher_matches_fn<Matcher>{m_Matcher},
83 info);
84 }
85
86 template <typename Return, typename... Args>
87 static constexpr void consume([[maybe_unused]] const call::Info<Return, Args...>& info) noexcept
88 {
89 }
90
91 [[nodiscard]]
92 StringT describe() const
93 {
94 return std::invoke(
95 m_DescribeStrategy,
96 detail::describe_hook::describe(m_Matcher));
97 }
98
99 private:
100 Matcher m_Matcher;
101 [[no_unique_address]] MatchesStrategy m_MatchesStrategy;
102 [[no_unique_address]] DescribeStrategy m_DescribeStrategy;
103 };
104}

135 template <
136 std::size_t... indices,
137 typename Matcher,
138 typename... Projections>
139 [[nodiscard]]
140 constexpr auto make_args_policy(
141 Matcher&& matcher,
142 std::tuple<Projections...>&& projections)
143 {
144 static_assert(
145 sizeof...(indices) == sizeof...(Projections),
146 "Indices and projections size mismatch.");
147
148 using arg_selector_t = mimicpp::detail::args_selector_fn<
149 std::add_lvalue_reference_t,
150 std::index_sequence<indices...>>;
151 using apply_strategy_t = mimicpp::detail::arg_list_indirect_apply_fn<std::remove_cvref_t<Projections>...>;
152 using describe_strategy_t = arg_requirement_describer<indices...>;
153
154 return expectation_policies::ArgsRequirement{
155 std::forward<Matcher>(matcher),
156 mimicpp::detail::apply_args_fn(
157 arg_selector_t{},
158 apply_strategy_t{std::move(projections)}),
159 describe_strategy_t{}};
160 }
161 }

222 template <
223 std::size_t first,
224 std::size_t... others,
225 typename Matcher,
226 typename... Projections>
227 [[nodiscard]]
228 constexpr auto args(Matcher&& matcher, Projections&&... projections)
229 noexcept(
230 std::is_nothrow_constructible_v<std::remove_cvref_t<Matcher>, Matcher>
231 && (... && std::is_nothrow_move_constructible_v<std::remove_cvref_t<Projections>>))
232 {
233 static_assert(
234 sizeof...(projections) <= 1u + sizeof...(others),
235 "The projection count exceeds the amount of indices.");
236
237 return detail::make_args_policy<first, others...>(
238 std::forward<Matcher>(matcher),
239 util::detail::expand_tuple<std::identity, 1u + sizeof...(others)>(
240 std::forward_as_tuple(std::forward<Projections>(projections)...)));
241 }
