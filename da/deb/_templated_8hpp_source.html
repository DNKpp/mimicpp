30 template <typename T, print_iterator OutIter>
31 constexpr OutIter pretty_template_name(OutIter out)
32 {
33 StringT name = type_name<T>();
34 auto const iter = std::ranges::find(name, '<');
35 MIMICPP_ASSERT(iter != name.cend(), "Given name is not a template.");
36 name.erase(iter, name.end());
37
38 return type::prettify_identifier(std::move(out), std::move(name));
39 }

104 template <template <typename...> typename Template, typename ArgList>
105 struct template_type_name_generator_fn
106 {
107 template <print_iterator OutIter>
108 constexpr OutIter operator()(OutIter out) const
109 {
110 using MinimalArgList = typename drop_default_args_for<Template, ArgList>::type;
111 using Type = util::type_list_populate_t<Template, MinimalArgList>;
112
113 out = pretty_template_name<Type>(std::move(out));
114 out = format::format_to(std::move(out), "<");
115 out = print_separated(std::move(out), ", ", MinimalArgList{});
116 out = format::format_to(std::move(out), ">");
117
118 return out;
119 }
120 };

141 template <template <typename...> typename Template, typename... Ts>
142 concept std_pmr_container =
143 requires {
144 typename Template<Ts...>::allocator_type;
145 typename Template<Ts...>::value_type;
146 }
147 && is_pmr_allocator<typename Template<Ts...>::allocator_type>::value
148 && default_arg_for<std::allocator<typename Template<Ts...>::value_type>, Template, util::type_list_pop_back_t<util::type_list<Ts...>>>;

150 static constexpr StringViewT stdPrefix{"std::"};
151
152 template <template <typename...> typename Template, typename ArgList>
153 struct potential_pmr_container_type_name_generator_fn
154 {
155 template <print_iterator OutIter>
156 OutIter operator()(OutIter out) const
157 {
158 StringT const name = std::invoke(
159 [] {
160 StringStreamT stream{};
161 std::invoke(
162 template_type_name_generator_fn<
163 Template,
164 util::type_list_pop_back_t<ArgList>>{},
165 std::ostreambuf_iterator{stream});
166 return std::move(stream).str();
167 });
168
169 // we do not want to accidentally manipulate non-std types, so make sure the `std::`` is actually part of the type
170 if (name.starts_with(stdPrefix))
171 {
172 out = format::format_to(std::move(out), "std::pmr::");
173 out = std::ranges::copy(
174 name.cbegin() + stdPrefix.size(),
175 name.cend(),
176 std::move(out))
177 .out;
178 }
179 // It's not an actual `std` type, but we've removed the allocator for the name generation, thus we need
180 // generate it again with the actual allocator.
181 else
182 {
183 out = std::invoke(
184 template_type_name_generator_fn<Template, ArgList>{},
185 std::move(out));
186 }
187
188 return out;
189 }
190 };
