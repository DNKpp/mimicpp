112 inline StringT stringify_call_report(const CallReport& report)
113 {
114 StringStreamT out{};
115 format_to(
116 std::ostreambuf_iterator{out},
117 "call from {}\n",
118 mimicpp::print(report.fromLoc));
119
120 format_to(
121 std::ostreambuf_iterator{out},
122 "constness: {}\n"
123 "value category: {}\n"
124 "return type: {}\n",
125 report.fromConstness,
126 report.fromCategory,
127 report.returnTypeIndex.name());
128
129 if (!std::ranges::empty(report.argDetails))
130 {
131 out << "args:\n";
132 for (const std::size_t i : std::views::iota(0u, std::ranges::size(report.argDetails)))
133 {
134 format_to(
135 std::ostreambuf_iterator{out},
136 "\targ[{}]: {{\n"
137 "\t\ttype: {},\n"
138 "\t\tvalue: {}\n"
139 "\t}},\n",
140 i,
141 report.argDetails[i].typeIndex.name(),
142 report.argDetails[i].stateString);
143 }
144 }
145
146 return std::move(out).str();

181 inline StringT stringify_expectation_report(const ExpectationReport& report)
182 {
183 StringStreamT out{};
184
185 out << "Expectation report:\n";
186
187 if (report.sourceLocation)
188 {
189 out << "from: ";
190 mimicpp::print(
191 std::ostreambuf_iterator{out},
192 *report.sourceLocation);
193 out << "\n";
194 }
195
196 if (report.timesDescription)
197 {
198 format_to(
199 std::ostreambuf_iterator{out},
200 "times: {}\n",
201 *report.timesDescription);
202 }
203
204 if (std::ranges::any_of(
205 report.expectationDescriptions,
206 [](const auto& desc) { return desc.has_value(); }))
207 {
208 out << "expects:\n";
209 for (const auto& desc
210 : report.expectationDescriptions
211 | std::views::filter([](const auto& desc) { return desc.has_value(); }))
212 {
213 format_to(
214 std::ostreambuf_iterator{out},
215 "\t{},\n",
216 *desc);
217 }
218 }
219
220 if (report.finalizerDescription)
221 {
222 format_to(
223 std::ostreambuf_iterator{out},
224 "finally: {}\n",
225 *report.finalizerDescription);
226 }
227
228 return std::move(out).str();

303 [[nodiscard]]
304 inline MatchResult evaluate_match_report(const MatchReport& report) noexcept
305 {
306 if (!std::ranges::all_of(report.expectationReports, &MatchReport::Expectation::isMatching))
307 {
308 return MatchResult::none;
309 }
310
311 if (!report.timesReport.isApplicable)
312 {
313 return MatchResult::inapplicable;
314 }
315
316 return MatchResult::full;
317 }

325 [[nodiscard]]
326 inline StringT stringify_match_report(const MatchReport& report)
327 {
328 std::vector<StringT> matchedExpectationDescriptions{};
329 std::vector<StringT> unmatchedExpectationDescriptions{};
330
331 for (const auto& [isMatching, description] : report.expectationReports)
332 {
333 if (description)
334 {
335 if (isMatching)
336 {
337 matchedExpectationDescriptions.emplace_back(*description);
338 }
339 else
340 {
341 unmatchedExpectationDescriptions.emplace_back(*description);
342 }
343 }
344 }
345
346 StringStreamT out{};
347
348 switch (evaluate_match_report(report))
349 {
350 case MatchResult::full:
351 out << "Matched expectation: {\n";
352 break;
353
354 case MatchResult::inapplicable:
355 format_to(
356 std::ostreambuf_iterator{out},
357 "Inapplicable, but otherwise matched expectation: {{\n"
358 "reason: {}\n",
359 report.timesReport.description.value_or("No reason provided."));
360 break;
361
362 case MatchResult::none:
363 out << "Unmatched expectation: {\n";
364 break;
365
366 // GCOVR_EXCL_START
367 default: // NOLINT(clang-diagnostic-covered-switch-default)
368 unreachable();
369 // GCOVR_EXCL_STOP
370 }
371
372 if (report.sourceLocation)
373 {
374 out << "from: ";
375 mimicpp::print(
376 std::ostreambuf_iterator{out},
377 *report.sourceLocation);
378 out << "\n";
379 }
380
381 if (!std::ranges::empty(unmatchedExpectationDescriptions))
382 {
383 out << "failed:\n";
384 for (const auto& desc : unmatchedExpectationDescriptions)
385 {
386 format_to(
387 std::ostreambuf_iterator{out},
388 "\t{},\n",
389 desc);
390 }
391 }
392
393 if (!std::ranges::empty(matchedExpectationDescriptions))
394 {
395 out << "passed:\n";
396 for (const auto& desc : matchedExpectationDescriptions)
397 {
398 format_to(
399 std::ostreambuf_iterator{out},
400 "\t{},\n",
401 desc);
402 }
403 }
404
405 out << "}\n";
406
407 return std::move(out).str();
408 }

586 template <
587 std::invocable<const StringT&> auto successReporter,
588 std::invocable<const StringT&> auto warningReporter,
589 std::invocable<const StringT&> auto failReporter
590 >
591 class BasicReporter
592 : public IReporter
593 {
594 public:
595 [[noreturn]]
596 void report_no_matches(const CallReport call, const std::vector<MatchReport> matchReports) override
597 {
598 StringStreamT ss{};
599 format_to(
600 std::ostreambuf_iterator{ss},
601 "No match for {}\n",
602 stringify_call_report(call));
603
604 if (std::ranges::empty(matchReports))
605 {
606 ss << "No expectations available.\n";
607 }
608 else
609 {
610 format_to(
611 std::ostreambuf_iterator{ss},
612 "{} available expectation(s):\n",
613 std::ranges::size(matchReports));
614
615 for (const auto& report : matchReports)
616 {
617 ss << stringify_match_report(report) << "\n";
618 }
619 }
620
621 send_fail(ss.str());
622 }
623
624 [[noreturn]]
625 void report_inapplicable_matches(const CallReport call, const std::vector<MatchReport> matchReports) override
626 {
627 StringStreamT ss{};
628 format_to(
629 std::ostreambuf_iterator{ss},
630 "No applicable match for {}\n",
631 stringify_call_report(call));
632
633 ss << "Tested expectations:\n";
634 for (const auto& report : matchReports)
635 {
636 ss << stringify_match_report(report) << "\n";
637 }
638
639 send_fail(ss.str());
640 }
641
642 void report_full_match(const CallReport call, const MatchReport matchReport) noexcept override
643 {
644 StringStreamT ss{};
645 format_to(
646 std::ostreambuf_iterator{ss},
647 "Found match for {}\n",
648 stringify_call_report(call));
649
650 ss << stringify_match_report(matchReport) << "\n";
651
652 send_success(ss.str());
653 }
654
655 void report_unfulfilled_expectation(const ExpectationReport expectationReport) override
656 {
657 if (0 == std::uncaught_exceptions())
658 {
659 StringStreamT ss{};
660 ss << "Unfulfilled expectation:\n"
661 << stringify_expectation_report(expectationReport) << "\n";
662
663 send_fail(ss.str());
664 }
665 }
666
667 void report_error(const StringT message) override
668 {
669 if (0 == std::uncaught_exceptions())
670 {
671 send_fail(message);
672 }
673 }
674
675 void report_unhandled_exception(
676 const CallReport call,
677 const ExpectationReport expectationReport,
678 const std::exception_ptr exception
679 ) override
680 {
681 StringStreamT ss{};
682 ss << "Unhandled exception: ";
683
684 try
685 {
686 std::rethrow_exception(exception);
687 }
688 catch (const std::exception& e)
689 {
690 format_to(
691 std::ostreambuf_iterator{ss},
692 "what: {}\n",
693 e.what());
694 }
695 catch (...)
696 {
697 ss << "Unknown exception type.\n";
698 }
699
700 format_to(
701 std::ostreambuf_iterator{ss},
702 "while checking expectation:\n"
703 "{}\n",
704 stringify_expectation_report(expectationReport));
705
706 format_to(
707 std::ostreambuf_iterator{ss},
708 "For {}\n",
709 stringify_call_report(call));
710
711 send_warning(ss.str());
712 }
713
714 private:
715 void send_success(const StringT& msg)
716 {
717 std::invoke(successReporter, msg);
718 }
719
720 void send_warning(const StringT& msg)
721 {
722 std::invoke(warningReporter, msg);
723 }

728 // GCOVR_EXCL_START
729 std::invoke(failReporter, msg);
730 unreachable();
731 // GCOVR_EXCL_STOP
732 }
