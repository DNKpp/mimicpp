51 template <typename T>
52 StringT type_name()
53 {
54 auto* const rawName = typeid(T).name();
55
56 // see: https://gcc.gnu.org/onlinedocs/libstdc++/manual/ext_demangling.html
57 int status{};
58 using free_deleter_t = decltype([](char* c) noexcept { std::free(c); });
59 std::unique_ptr<char, free_deleter_t> const demangledName{
60 abi::__cxa_demangle(rawName, nullptr, nullptr, &status)};
61 if (0 == status)
62 {
63 return {demangledName.get()};
64 }
65
66 return {rawName};
67 }
68}
