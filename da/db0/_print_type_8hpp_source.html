99 template <typename T>
100 StringT type_name()
101 {
102 auto* const rawName = typeid(T).name();
103
104 // see: https://gcc.gnu.org/onlinedocs/libstdc++/manual/ext_demangling.html
105 int status{};
106 using free_deleter_t = decltype([](char* c) noexcept { std::free(c); });
107 std::unique_ptr<char, free_deleter_t> const demangledName{
108 abi::__cxa_demangle(rawName, nullptr, nullptr, &status)};
109 if (0 == status)
110 {
111 return {demangledName.get()};
112 }
113
114 return {rawName};
115 }

143 parser.parse_type();

150 [[nodiscard]]
151 MIMICPP_DETAIL_CONSTEXPR_STRING StringT remove_template_details(StringT name)
152 {
153 if (name.ends_with(']'))
154 {
155 auto rest = name | std::views::reverse | std::views::drop(1);
156 if (auto const closingIter = util::find_closing_token(rest, ']', '[');
157 closingIter != rest.end())
158 {
159 auto const end = std::ranges::find_if_not(
160 closingIter + 1,
161 rest.end(),
162 lexing::is_space);
163 name.erase(end.base(), name.end());
164 }
165 }
166
167 return name;
168 }
169 }

171 template <print_iterator OutIter>
172 MIMICPP_DETAIL_CONSTEXPR_STRING OutIter prettify_function(OutIter out, StringT name)
173 {
174 name = detail::remove_template_details(std::move(name));
175
176 static_assert(parsing::parser_visitor<PrintVisitor<OutIter>>);
177
178 PrintVisitor visitor{std::move(out)};
179 parsing::NameParser parser{std::ref(visitor), name};
180 parser.parse_function();
181
182 return visitor.out();
183 }
