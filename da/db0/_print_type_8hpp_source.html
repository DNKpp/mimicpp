37 template <typename T>
38 StringT type_name()
39 {
40 StringT name{typeid(T).name()};
41
42 // see: https://gcc.gnu.org/onlinedocs/libstdc++/manual/ext_demangling.html
43 int status{};
44 using free_deleter_t = decltype([](char* c) noexcept { std::free(c); });
45 std::unique_ptr<char, free_deleter_t> const demangledName{
46 abi::__cxa_demangle(name.data(), nullptr, nullptr, &status)};
47 if (0 == status)
48 {
49 name.assign(demangledName.get());
50 }
51
52 return name;
53 }

55 inline StringT prettify_type_name(StringT name)
56 {
57 static const RegexT unifyClosingAngleBrackets{R"(>\s*>)"};
58 name = util::regex_replace_all(std::move(name), unifyClosingAngleBrackets, StringViewT{">>"});

61 static const RegexT handleStdNamespace{"std::__1::"};
62 #else
63 static const RegexT handleStdNamespace{"std::__cxx11::"};
64 #endif
65 name = std::regex_replace(name, handleStdNamespace, "std::");
66
67 static const RegexT omitAnonymousNamespace{R"(\‍(anonymous namespace\)::)"};
68 name = std::regex_replace(name, omitAnonymousNamespace, "");
69
70 return name;

84 inline StringT prettify_type_name(StringT name)
85 {
86 static const RegexT unifyClosingAngleBrackets{R"(>\s*>)"};
87 name = util::regex_replace_all(std::move(name), unifyClosingAngleBrackets, StringViewT{">>"});
88
89 static const RegexT unifyComma{R"(\s*,\s*)"};
90 name = std::regex_replace(name, unifyComma, ", ");
91
92 static const RegexT omitClassStructEnum{R"(\b(class|struct|enum)\s+)"};
93 name = std::regex_replace(name, omitClassStructEnum, "");
94

96 name = std::regex_replace(name, omitAnonymousNamespace, "");
