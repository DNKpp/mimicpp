95 StringT type_name()
96 {
97 auto* const rawName = typeid(T).name();

99 // see: https://gcc.gnu.org/onlinedocs/libstdc++/manual/ext_demangling.html
100 int status{};
101 using free_deleter_t = decltype([](char* c) noexcept { std::free(c); });
102 std::unique_ptr<char, free_deleter_t> const demangledName{
103 abi::__cxa_demangle(rawName, nullptr, nullptr, &status)};
104 if (0 == status)
105 {
106 return {demangledName.get()};
107 }
108
109 return {rawName};
110 }

143 namespace detail

145 [[nodiscard]]
146 constexpr StringT remove_template_details(StringT name)
147 {
148 if (name.ends_with(']'))
149 {
150 auto rest = name | std::views::reverse | std::views::drop(1);
151 if (auto const closingIter = util::find_closing_token(rest, ']', '[');
152 closingIter != rest.end())
153 {
154 auto const end = std::ranges::find_if_not(
155 closingIter + 1,
156 rest.end(),
157 lexing::is_space);
158 name.erase(end.base(), name.end());
159 }
160 }
161
162 return name;
163 }
164 }

166 template <print_iterator OutIter>
167 constexpr OutIter prettify_function(OutIter out, StringT name)
168 {
169 name = detail::remove_template_details(std::move(name));
170
171 static_assert(parsing::parser_visitor<PrintVisitor<OutIter>>);
172
173 PrintVisitor visitor{std::move(out)};
174 parsing::NameParser parser{std::ref(visitor), name};
175 parser.parse_function();
176
177 return visitor.out();
178 }
