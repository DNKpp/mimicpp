106 template <typename T>
107 concept backend =
108 std::copyable<T>
109 && requires(backend_traits<std::remove_cvref_t<T>> traits, const std::remove_cvref_t<T>& backend, const std::size_t value) {
110 { decltype(traits)::current(value) } -> std::convertible_to<std::remove_cvref_t<T>>;
111 { decltype(traits)::size(backend) } -> std::convertible_to<std::size_t>;
112 { decltype(traits)::empty(backend) } -> std::convertible_to<bool>;
113 { decltype(traits)::description(backend, value) } -> std::convertible_to<std::string>;
114 { decltype(traits)::source_file(backend, value) } -> std::convertible_to<std::string>;
115 { decltype(traits)::source_line(backend, value) } -> std::convertible_to<std::size_t>;
116 };

195 template <typename Inner>
196 requires(!std::same_as<Stacktrace, std::remove_cvref_t<Inner>>)
197 && stacktrace::backend<Inner>
198 [[nodiscard]]
199 explicit constexpr Stacktrace(Inner&& inner)
200 : m_Inner{std::forward<Inner>(inner)},
201 m_SizeFn{&stacktrace::detail::size<std::remove_cvref_t<Inner>>},
202 m_EmptyFn{&stacktrace::detail::empty<std::remove_cvref_t<Inner>>},
203 m_DescriptionFn{&stacktrace::detail::description<std::remove_cvref_t<Inner>>},
204 m_SourceFileFn{&stacktrace::detail::source_file<std::remove_cvref_t<Inner>>},
205 m_SourceLineFn{&stacktrace::detail::source_line<std::remove_cvref_t<Inner>>}
206 {
207 }

212 [[nodiscard]]
213 Stacktrace(const Stacktrace&) = default;
214
218 Stacktrace& operator=(const Stacktrace&) = default;
219
223 [[nodiscard]]
224 Stacktrace(Stacktrace&&) = default;
225
229 Stacktrace& operator=(Stacktrace&&) = default;
230
235 [[nodiscard]]
236 constexpr std::size_t size() const
237 {
238 return std::invoke(m_SizeFn, m_Inner);
239 }
240
245 [[nodiscard]]
246 constexpr bool empty() const
247 {
248 return std::invoke(m_EmptyFn, m_Inner);
249 }
250
256 [[nodiscard]]
257 constexpr std::string description(const std::size_t at) const
258 {
259 return std::invoke(m_DescriptionFn, m_Inner, at);
260 }
261
267 [[nodiscard]]
268 constexpr std::string source_file(const std::size_t at) const
269 {
270 return std::invoke(m_SourceFileFn, m_Inner, at);
271 }
272
278 [[nodiscard]]
279 constexpr std::size_t source_line(const std::size_t at) const
280 {
281 return std::invoke(m_SourceLineFn, m_Inner, at);
282 }
283
284 [[nodiscard]]
285 friend bool operator==(const Stacktrace& lhs, const Stacktrace& rhs)
286 {
287 return lhs.size() == rhs.size()
288 && std::ranges::all_of(
289 std::views::iota(0u, lhs.size()),
290 [&](const std::size_t index) {
291 return lhs.description(index) == rhs.description(index)
292 && lhs.source_file(index) == rhs.source_file(index)
293 && lhs.source_line(index) == rhs.source_line(index);
294 });
295 }
296
297 private:
298 std::any m_Inner;
299 size_fn m_SizeFn;
300 empty_fn m_EmptyFn;
301 description_fn m_DescriptionFn;
302 source_file_fn m_SourceFileFn;
303 source_line_fn m_SourceLineFn;

381 template <print_iterator OutIter>
382 OutIter print_entry(OutIter out, Stacktrace const& stacktrace, std::size_t const index)
383 {
384 assert(index < stacktrace.size() && "Index out of bounds.");
385
386 out = format::format_to(std::move(out), "`");
387 out = print_path(std::move(out), stacktrace.source_file(index));
388 out = format::format_to(std::move(out), "`");
389 out = format::format_to(
390 std::move(out),
391 "#L{}, `{}`",
392 stacktrace.source_line(index),
393 stacktrace.description(index));
394
395 return out;
396 }

399template <>
400struct mimicpp::printing::detail::state::common_type_printer<mimicpp::Stacktrace>
401{
402 template <print_iterator OutIter>
403 static OutIter print(OutIter out, const Stacktrace& stacktrace)
404 {
405 if (stacktrace.empty())
406 {
407 return format::format_to(std::move(out), "empty");
408 }
409
410 for (const std::size_t i : std::views::iota(0u, stacktrace.size()))
411 {
412 out = format::format_to(std::move(out), "#{} ", i);
413 out = stacktrace::detail::print_entry(
414 std::move(out),
415 stacktrace,
416 i);
417 out = format::format_to(std::move(out), "\n");
418 }
419
420 return out;
421 }
