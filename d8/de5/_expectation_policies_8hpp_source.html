21 [[nodiscard]]
22 inline StringT describe_times_state(const std::size_t current, const std::size_t min, const std::size_t max)
23 {
24 const auto verbalizeValue = [](const std::size_t value)-> StringT
25 {
26 switch (value)
27 {
28 case 0:
29 return "never";
30 case 1:
31 return "once";
32 case 2:
33 return "twice";
34 default:
35 return format::format("{} times", value);
36 }
37 };
38
39 if (current == max)
40 {
41 return format::format(
42 "inapplicable: already saturated (matched {})",
43 verbalizeValue(current));
44 }
45
46 if (min <= current)
47 {
48 return format::format(
49 "applicable: accepts further matches (matched {} out of {} times)",
50 current,
51 max);
52 }
53
54 const auto verbalizeInterval = [verbalizeValue](const std::size_t start, const std::size_t end)
55 {
56 if (start < end)
57 {
58 return format::format(
59 "between {} and {} times",
60 start,
61 end);
62 }
63
64 return format::format(
65 "exactly {}",
66 verbalizeValue(end));
67 };
68
69 return format::format(
70 "unsatisfied: matched {} - {} is expected",
71 verbalizeValue(current),
72 verbalizeInterval(min, max));
73 }
74}

413 template <typename Action, template <typename> typename Projection>
414 requires std::same_as<Action, std::remove_cvref_t<Action>>
415 class ApplyAllArgsAction
416 {
417 public:
418 [[nodiscard]]
419 explicit constexpr ApplyAllArgsAction(
420 Action action = Action{}
421 ) noexcept(std::is_nothrow_move_constructible_v<Action>)
422 : m_Action{std::move(action)}
423 {
424 }
425
426 template <typename Arg>
427 using ProjectedArgT = Projection<Arg>;
428
429 template <typename Return, typename... Args>
430 requires std::invocable<
431 const Action&,
432 ProjectedArgT<Args>...>
433 constexpr decltype(auto) operator ()(
434 const call::Info<Return, Args...>& callInfo
435 ) const noexcept(
436 std::is_nothrow_invocable_v<
437 const Action&,
438 ProjectedArgT<Args>...>)
439 {
440 static_assert(
441 (... && explicitly_convertible_to<Args&, ProjectedArgT<Args>>),
442 "Projection can not be applied.");
443
444 return std::apply(
445 [this](auto&... args) -> decltype(auto)
446 {
447 return std::invoke(
448 m_Action,
449 static_cast<ProjectedArgT<Args>>(
450 args.get())...);
451 },
452 callInfo.args);
453 }
454
455 private:
456 Action m_Action;

459 template <typename Action, template <typename> typename Projection, std::size_t... indices>
460 requires std::same_as<Action, std::remove_cvref_t<Action>>
461 class ApplyArgsAction
462 {
463 public:
464 [[nodiscard]]
465 explicit constexpr ApplyArgsAction(
466 Action action = Action{}
467 ) noexcept(std::is_nothrow_move_constructible_v<Action>)
468 : m_Action{std::move(action)}
469 {
470 }
471
472 template <std::size_t index, typename... Args>
473 using ArgListElementT = std::tuple_element_t<index, std::tuple<Args...>>;
474
475 template <std::size_t index, typename... Args>
476 using ProjectedArgListElementT = Projection<ArgListElementT<index, Args...>>;
477
478 template <typename Return, typename... Args>
479 requires (... && (indices < sizeof...(Args)))
480 && std::invocable<
481 const Action&,
482 ProjectedArgListElementT<indices, Args...>...>
483 constexpr decltype(auto) operator ()(
484 const call::Info<Return, Args...>& callInfo
485 ) const noexcept(
486 std::is_nothrow_invocable_v<
487 const Action&,
488 ProjectedArgListElementT<indices, Args...>...>)
489 {
490 static_assert(
491 (explicitly_convertible_to<
492 ArgListElementT<indices, Args...>&,
493 ProjectedArgListElementT<indices, Args...>>
494 && ...),
495 "Projection can not be applied.");
496
497 return std::invoke(
498 m_Action,
499 static_cast<ProjectedArgListElementT<indices, Args...>>(
500 std::get<indices>(callInfo.args).get())...);
