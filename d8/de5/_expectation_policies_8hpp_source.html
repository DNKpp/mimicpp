185 [[nodiscard]]

195 }

199 {
200 if (const std::optional<StringT> description = m_Matcher.describe())
201 {
202 return std::invoke(

268 template <typename Action, template <typename> typename Projection>
269 requires std::same_as<Action, std::remove_cvref_t<Action>>
270 class ApplyAllArgsAction
271 {
272 public:
273 [[nodiscard]]
274 explicit constexpr ApplyAllArgsAction(
275 Action action = Action{}
276 ) noexcept(std::is_nothrow_move_constructible_v<Action>)
277 : m_Action{std::move(action)}
278 {
279 }
280
281 template <typename Arg>
282 using ProjectedArgT = Projection<Arg>;
283
284 template <typename Return, typename... Args>
285 requires std::invocable<
286 const Action&,
287 ProjectedArgT<Args>...>
288 constexpr decltype(auto) operator ()(
289 const call::Info<Return, Args...>& callInfo
290 ) const noexcept(
291 std::is_nothrow_invocable_v<
292 const Action&,
293 ProjectedArgT<Args>...>)
294 {
295 static_assert(
296 (... && explicitly_convertible_to<Args&, ProjectedArgT<Args>>),
297 "Projection can not be applied.");
298
299 return std::apply(
300 [this](auto&... args) -> decltype(auto)
301 {
302 return std::invoke(
303 m_Action,
304 static_cast<ProjectedArgT<Args>>(
305 args.get())...);
306 },
307 callInfo.args);
308 }
309
310 private:
311 Action m_Action;

314 template <typename Action, template <typename> typename Projection, std::size_t... indices>
315 requires std::same_as<Action, std::remove_cvref_t<Action>>
316 class ApplyArgsAction
317 {
318 public:
319 [[nodiscard]]
320 explicit constexpr ApplyArgsAction(
321 Action action = Action{}
322 ) noexcept(std::is_nothrow_move_constructible_v<Action>)
323 : m_Action{std::move(action)}
324 {
325 }
326
327 template <std::size_t index, typename... Args>
328 using ArgListElementT = std::tuple_element_t<index, std::tuple<Args...>>;
329
330 template <std::size_t index, typename... Args>
331 using ProjectedArgListElementT = Projection<ArgListElementT<index, Args...>>;
332
333 template <typename Return, typename... Args>
334 requires (... && (indices < sizeof...(Args)))
335 && std::invocable<
336 const Action&,
337 ProjectedArgListElementT<indices, Args...>...>
338 constexpr decltype(auto) operator ()(
339 const call::Info<Return, Args...>& callInfo
340 ) const noexcept(
341 std::is_nothrow_invocable_v<
342 const Action&,
343 ProjectedArgListElementT<indices, Args...>...>)
344 {
345 static_assert(
346 (explicitly_convertible_to<
347 ArgListElementT<indices, Args...>&,
348 ProjectedArgListElementT<indices, Args...>>
349 && ...),
350 "Projection can not be applied.");
351
352 return std::invoke(
353 m_Action,
354 static_cast<ProjectedArgListElementT<indices, Args...>>(
355 std::get<indices>(callInfo.args).get())...);
356 }
357
358 private:
359 Action m_Action;
360 };
