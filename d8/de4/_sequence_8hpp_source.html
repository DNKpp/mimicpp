83 template <typename Id, auto priorityStrategy>
84 requires std::is_enum_v<Id>
85 && std::signed_integral<std::underlying_type_t<Id>>
86 && std::convertible_to<
87 std::invoke_result_t<decltype(priorityStrategy), Id, int>,
88 int>
89 class BasicSequence
90 {
91 public:
92 using IdT = Id;
93
94 ~BasicSequence() noexcept(false)
95 {
96 const auto iter = std::ranges::find_if_not(
97 m_Entries.cbegin() + m_Cursor,
98 m_Entries.cend(),
99 [](const State s) noexcept {
100 return s == State::satisfied
101 || s == State::saturated;
102 });
103
104 if (iter != m_Entries.cend())
105 {
106 reporting::detail::report_error(
107 format::format(
108 "Unfulfilled sequence. {} out of {} expectation(s) are satisfied.",
109 std::ranges::distance(m_Entries.cbegin(), iter),
110 m_Entries.size()));
111 }
112 }
113
114 [[nodiscard]]
115 BasicSequence() = default;
116
117 BasicSequence(const BasicSequence&) = delete;
118 BasicSequence& operator=(const BasicSequence&) = delete;
119 BasicSequence(BasicSequence&&) = delete;
120 BasicSequence& operator=(BasicSequence&&) = delete;

122 [[nodiscard]]
123 constexpr std::optional<int> priority_of(const IdT id) const noexcept
124 {
125 MIMICPP_ASSERT(is_valid(id), "Invalid id given.");
126
127 if (is_consumable(id))
128 {
129 return std::invoke(
130 priorityStrategy,
131 id,
132 m_Cursor);
133 }
134
135 return std::nullopt;
136 }
137
138 constexpr void set_satisfied(const IdT id) noexcept
139 {
140 MIMICPP_ASSERT(is_valid(id), "Invalid id given.");
141 MIMICPP_ASSERT(m_Cursor <= util::to_underlying(id), "Invalid state.");
142
143 auto& element = m_Entries[util::to_underlying(id)];
144 MIMICPP_ASSERT(element == State::unsatisfied, "Element is in unexpected state.");
145 element = State::satisfied;
146 }
147
148 constexpr void set_saturated(const IdT id) noexcept
149 {
150 MIMICPP_ASSERT(is_valid(id), "Invalid id given.");
151 const auto index = util::to_underlying(id);
152 MIMICPP_ASSERT(m_Cursor <= index, "Invalid state.");
153
154 auto& element = m_Entries[index];
155 MIMICPP_ASSERT(element == State::unsatisfied || element == State::satisfied, "Element is in unexpected state.");
156 element = State::saturated;
157 }
158
159 [[nodiscard]]
160 constexpr bool is_consumable(const IdT id) const noexcept
161 {
162 MIMICPP_ASSERT(is_valid(id), "Invalid id given.");
163
164 const int index = util::to_underlying(id);
165 const auto state = m_Entries[index];
166 return m_Cursor <= index
167 && std::ranges::all_of(
168 m_Entries.begin() + m_Cursor,
169 m_Entries.begin() + index,
170 [](const State s) noexcept {
171 return s == State::satisfied
172 || s == State::saturated;
173 })
174 && (state == State::unsatisfied
175 || state == State::satisfied);
176 }
177
178 constexpr void consume(const IdT id) noexcept
179 {
180 MIMICPP_ASSERT(is_consumable(id), "Sequence is not in consumable state.");
181
182 m_Cursor = util::to_underlying(id);
183 }
184
185 [[nodiscard]]
186 constexpr IdT add()
187 {
188 if (!std::in_range<std::underlying_type_t<IdT>>(m_Entries.size()))
189 [[unlikely]]
190 {
191 throw std::runtime_error{
192 "Sequence already holds maximum amount of elements."};
193 }
194
195 m_Entries.emplace_back(State::unsatisfied);
196 return static_cast<IdT>(m_Entries.size() - 1);
197 }
198
199 [[nodiscard]]
200 constexpr Tag tag() const noexcept
201 {
202 return Tag{
203 std::bit_cast<std::ptrdiff_t>(this)};
204 }
205
206 private:
207 enum class State
208 {
209 unsatisfied,
210 satisfied,
211 saturated
212 };
213
214 std::vector<State> m_Entries{};
215 int m_Cursor{};
216
217 [[nodiscard]]
218 constexpr bool is_valid(const IdT id) const noexcept
219 {
220 return 0 <= util::to_underlying(id)
221 && std::cmp_less(util::to_underlying(id), m_Entries.size());
222 }

225 class LazyStrategy
226 {
227 public:
228 [[nodiscard]]
229 constexpr int operator()(const auto id, const int cursor) const noexcept
230 {
231 const auto index = util::to_underlying(id);
232 MIMICPP_ASSERT(std::cmp_less_equal(cursor, index), "Invalid state.");
233
234 return std::numeric_limits<int>::max()
235 - (static_cast<int>(index) - cursor);
236 }

393 [[nodiscard]]
394 constexpr bool has_better_rating(
395 const std::span<const rating> lhs,
396 const std::span<const rating> rhs) noexcept
397 {
398 int rating{};
399 for (const auto& [lhsPriority, lhsTag] : lhs)
400 {
401 if (const auto iter = std::ranges::find(rhs, lhsTag, &rating::tag);
402 iter != std::ranges::end(rhs))
403 {
404 rating += lhsPriority < iter->priority ? -1 : 1;
405 }
406 else
407 {
408 return true;
409 }
410 }
411
412 return 0 <= rating;
413 }
