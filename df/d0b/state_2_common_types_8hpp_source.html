95 template <typename T>
96 struct common_type_printer<T*>
97 {
98 template <print_iterator OutIter>
99 static OutIter print(OutIter out, T const* ptr)
100 {
101 if (auto const* inner = std::to_address(ptr))
102 {
103 auto const value = std::bit_cast<std::uintptr_t>(inner);
104 if constexpr (4u < sizeof(std::uintptr_t))
105 {
106 return format::format_to(
107 std::move(out),
108 "0x{:0>16x}",
109 value);
110 }
111 else
112 {
113 return format::format_to(
114 std::move(out),
115 "0x{:0>8x}",
116 value);
117 }
118 }
119
120 return mimicpp::print(std::move(out), nullptr);
121 }

150 template <string String>
151 struct common_type_printer<String>
152 {
153 template <std::common_reference_with<String> T, print_iterator OutIter>
154 static OutIter print(OutIter out, T& str)
155 {
156 if constexpr (constexpr auto prefix = string_literal_prefix<string_char_t<String>>;
157 !std::ranges::empty(prefix))
158 {
159 out = std::ranges::copy(prefix, std::move(out)).out;
160 }
161
162 out = format::format_to(std::move(out), "\"");
163
164 if constexpr (std::same_as<CharT, string_char_t<String>>)
165 {
166 out = std::ranges::copy(
167 string_traits<String>::view(str),
168 std::move(out))
169 .out;
170 }
171 // required for custom char types
172 else if constexpr (printer_for<custom::Printer<string_char_t<String>>, OutIter, string_char_t<String>>)
173 {
174 for (custom::Printer<string_char_t<String>> printer{};
175 const string_char_t<String>& c : string_traits<String>::view(str))
176 {
177 out = printer.print(std::move(out), c);
178 }
179 }
180 else
181 {
182 constexpr auto to_dump = [](const string_char_t<String>& c) noexcept {
183 using intermediate_t = uint_with_size_t<sizeof c>;
184 return std::bit_cast<intermediate_t>(c);
185 };
186
187 auto view = string_traits<std::remove_cvref_t<T>>::view(std::forward<T>(str));
188 auto iter = std::ranges::begin(view);
189 if (const auto end = std::ranges::end(view);
190 iter != end)
191 {
192 out = format::format_to(
193 std::move(out),
194 "{:#x}",
195 to_dump(*iter++));
196
197 for (; iter != end; ++iter)
198 {
199 out = format::format_to(
200 std::move(out),
201 ", {:#x}",
202 to_dump(*iter));
203 }
204 }
205 }
206
207 return format::format_to(std::move(out), "\"");
208 }
209 };
