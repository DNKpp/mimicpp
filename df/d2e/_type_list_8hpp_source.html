138 requires(sizeof...(Elements) <= n)
139 [[nodiscard]]
140 constexpr auto expand_tuple(std::tuple<Elements...>&& tuple)
141 {
142 // prior to c++23, tuple_cat does not officially support tuple-like types,
143 // thus we transform the generated array manually
144 return std::tuple_cat(
145 std::move(tuple),
146 std::apply(
147 [](auto&&... elements) { return std::make_tuple(std::move(elements)...); },
148 std::array<FillElement, n - sizeof...(Elements)>{}));
