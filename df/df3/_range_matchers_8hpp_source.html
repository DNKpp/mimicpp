26 // This is a small compatibility layer for gcc 11 and prior, because they do just provide a partial
27 // implementation of `std::views::all`. In fact, the `std::ranges::owning_view` impl is completely missing.
28 template <std::ranges::range Range>
29 [[nodiscard]]
30 constexpr auto store(Range&& range)
31 {
32 if constexpr (std::ranges::view<std::decay_t<Range>>)
33 {
34 return std::forward<Range>(range);
35 }
36 else if constexpr (requires { std::ranges::ref_view{std::declval<Range>()}; })
37 {
38 return std::ranges::ref_view{std::forward<Range>(range)};
39 }
40 else
41 {
42 return std::forward<Range>(range);
43 }
44 }
45
46#else
47 constexpr auto store = std::views::all;
48#endif
49 }
