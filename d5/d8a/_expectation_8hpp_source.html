36 template <typename Return, typename... Params, typename Signature>
37 [[nodiscard]]
38 std::optional<reporting::RequirementOutcomes> determine_requirement_outcomes(
39 reporting::TargetReport const& target,
40 call::Info<Return, Params...> const& call,
41 Expectation<Signature> const& expectation) noexcept
42 {
43 try
44 {
45 return expectation.matches(call);
46 }
47 catch (...)
48 {
49 reporting::detail::report_unhandled_exception(
50 reporting::make_call_report(target, call),
51 expectation.report(),
52 std::current_exception());
53 }
54
55 return std::nullopt;
56 }

58 [[nodiscard]]
59 std::vector<reporting::ExpectationReport> gather_expectation_reports(auto&& expectationPtrs)
60 {
61 auto view = expectationPtrs
62 | std::views::transform([](auto const& exp) { return exp->report(); });
63 return std::vector<reporting::ExpectationReport>{
64 view.begin(),
65 view.end()};

68 template <typename Signature>
69 [[nodiscard]]
70 std::vector<reporting::NoMatchReport> make_no_match_reports(
71 std::vector<std::tuple<Expectation<Signature>*, reporting::RequirementOutcomes>>&& outcomes)
72 {
73 std::vector<reporting::NoMatchReport> reports{};
74 reports.reserve(outcomes.size());
75 for (auto const& [expectationPtr, outcome] : outcomes)
76 {
77 reports.emplace_back(
78 expectationPtr->report(),
79 std::move(outcome));
80 }
81
82 return reports;
83 }

85 [[nodiscard]]
86 constexpr auto find_best_match(std::span<reporting::ExpectationReport const> const matches)
87 {
88 constexpr auto ratings = [](auto const& el) noexcept -> const auto& {
89 return std::get<reporting::state_applicable>(el.controlReport)
90 .sequenceRatings;
91 };
92
93 auto best = std::ranges::begin(matches);
94 for (auto iter = best + 1;
95 iter != std::ranges::end(matches);
96 ++iter)
97 {
98 if (!sequence::detail::has_better_rating(
99 ratings(*best),
100 ratings(*iter)))
101 {
102 best = iter;
103 }
104 }
105
106 return std::ranges::distance(std::ranges::begin(matches), best);
107 }

308 void push(std::shared_ptr<ExpectationT> expectation)
309 {
310 const std::scoped_lock lock{m_ExpectationsMx};
311
312 MIMICPP_ASSERT(std::ranges::find(m_Expectations, expectation) == std::ranges::end(m_Expectations), "Expectation already belongs to this storage.");
313
314 m_Expectations.emplace_back(std::move(expectation));

324 void remove(std::shared_ptr<ExpectationT> expectation)
325 {
326 const std::scoped_lock lock{m_ExpectationsMx};
327
328 auto iter = std::ranges::find(m_Expectations, expectation);
329 MIMICPP_ASSERT(iter != std::ranges::end(m_Expectations), "Expectation does not belong to this storage.");
330 m_Expectations.erase(iter);
331
332 if (!expectation->is_satisfied())
333 {
334 reporting::detail::report_unfulfilled_expectation(
335 expectation->report());
336 }
337 }

349 [[nodiscard]]
350 ReturnT handle_call(reporting::TargetReport target, CallInfoT call)
351 {
352 std::vector<ExpectationT*> matches{};
353 std::vector<ExpectationT*> inapplicableMatches{};
354 std::vector<std::tuple<ExpectationT*, reporting::RequirementOutcomes>> noMatches{};
355
356 std::scoped_lock const lock{m_ExpectationsMx};
357 evaluate_expectations(target, call, matches, inapplicableMatches, noMatches);
358
359 if (!std::ranges::empty(matches))
360 {
361 std::vector reports = detail::gather_expectation_reports(matches);
362 MIMICPP_ASSERT(matches.size() == reports.size(), "Size mismatch.");
363 auto const bestIndex = detail::find_best_match(reports);
364 MIMICPP_ASSERT(0 <= bestIndex && bestIndex < std::ssize(reports), "Invalid index.");
365
366 auto& report = reports[bestIndex];
367 auto& expectation = *matches[bestIndex];
368
369 // Todo: Avoid the call copy
370 // Maybe we can prevent the copy here, but we should keep the instruction order as-is, because
371 // in cases of a throwing finalizer, we might introduce bugs. At least there are some tests, which
372 // will fail if done wrong.
373 reporting::detail::report_full_match(
374 reporting::make_call_report(std::move(target), call),
375 std::move(report));
376 expectation.consume(call);
377 return expectation.finalize_call(call);
378 }
379
380 if (!std::ranges::empty(inapplicableMatches))
381 {
382 reporting::detail::report_inapplicable_matches(
383 reporting::make_call_report(std::move(target), std::move(call)),
384 detail::gather_expectation_reports(inapplicableMatches));
385 }
386
387 reporting::detail::report_no_matches(
388 reporting::make_call_report(std::move(target), std::move(call)),
389 detail::make_no_match_reports(std::move(noMatches)));
390 }
391
392 private:
393 std::vector<std::shared_ptr<ExpectationT>> m_Expectations{};
394 std::mutex m_ExpectationsMx{};

396 void evaluate_expectations(
397 reporting::TargetReport const& target,
398 CallInfoT const& call,
399 std::vector<ExpectationT*>& matches,
400 std::vector<ExpectationT*>& inapplicableMatches,
401 std::vector<std::tuple<ExpectationT*, reporting::RequirementOutcomes>>& noMatches)
402 {
403 for (auto const& exp : m_Expectations | std::views::reverse)
404 {
405 if (std::optional outcomes = detail::determine_requirement_outcomes(target, call, *exp))
406 {
407 if (std::ranges::any_of(outcomes->outcomes, [](auto const& el) { return el == false; }))
408 {
409 noMatches.emplace_back(exp.get(), *std::move(outcomes));
410 }
411 else if (!exp->is_applicable())
412 {
413 inapplicableMatches.emplace_back(exp.get());
414 }
415 else
416 {
417 matches.emplace_back(exp.get());
418 }
419 }
420 }
421 }

427 template <typename T, typename Signature>
428 concept expectation_policy_for = std::is_move_constructible_v<T>
429 && std::is_destructible_v<T>
430 && std::same_as<T, std::remove_cvref_t<T>>
431 && requires(T& policy, const call::info_for_signature_t<Signature>& info) {
432 { std::as_const(policy).is_satisfied() } noexcept -> util::boolean_testable;
433 { std::as_const(policy).matches(info) } -> util::boolean_testable;
434 { std::as_const(policy).describe() } -> util::explicitly_convertible_to<std::optional<StringT>>;
435 { policy.consume(info) };
436 };

441 template <typename T, typename Signature>
442 concept finalize_policy_for = std::is_move_constructible_v<T>
443 && std::is_destructible_v<T>
444 && std::same_as<T, std::remove_cvref_t<T>>
445 && requires(T& policy, const call::info_for_signature_t<Signature>& info) {
446 { policy.finalize_call(info) } -> std::convertible_to<signature_return_type_t<Signature>>;
447 };

453 concept control_policy = std::is_move_constructible_v<T>
454 && std::is_destructible_v<T>
455 && std::same_as<T, std::remove_cvref_t<T>>
456 && requires(T& policy) {
457 { std::as_const(policy).is_satisfied() } noexcept -> util::boolean_testable;

503 FinalizerArg&& finalizerArg,

505 noexcept(

510 m_Target{std::move(target)},
511 m_ControlPolicy{std::forward<ControlPolicyArg>(controlArg)},

521 reporting::ExpectationReport report() const override
522 {

524 .from = m_From,
525 .target = m_Target,
526 .controlReport = m_ControlPolicy.state(),
527 .finalizerDescription = std::nullopt,
528 .requirementDescriptions = std::apply(
529 [&](auto const&... policies) {
530 return std::vector<std::optional<StringT>>{
