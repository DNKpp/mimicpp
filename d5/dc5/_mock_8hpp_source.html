293 template <typename Signature, typename... Params>
294 class BasicMock<Signature, util::type_list<Params...>>
295 : public MockFrontend<
296 // MockFrontend doesn't need to know about the call-convention, thus remove it
297 BasicMock<Signature, util::type_list<Params...>>,
298 signature_remove_call_convention_t<Signature>>,
299 public call_interface_t<
300 BasicMock<Signature, util::type_list<Params...>>,
301 Signature>
302 {
303 using SignatureT = signature_remove_call_convention_t<Signature>;
304
305 friend class MockFrontend<BasicMock, SignatureT>;
306 friend call_interface_t<BasicMock, Signature>;
307
308 static constexpr Constness constQualification = signature_const_qualification_v<SignatureT>;
309 static constexpr ValueCategory refQualification = signature_ref_qualification_v<SignatureT>;
310
311 protected:
312 using ExpectationCollectionPtrT = expectation_collection_ptr_for<SignatureT>;
313
314 [[nodiscard]]
315 explicit BasicMock(
316 ExpectationCollectionPtrT collection,
317 MockSettings settings) noexcept
318 : m_Expectations{std::move(collection)},
319 m_Settings{std::move(settings)}
320 {
321 MIMICPP_ASSERT(m_Settings.name, "Empty mock-name.");
322
323 m_Settings.stacktraceSkip += 2u; // skips the operator() and the handle_call from the stacktrace
324 }
325
326 private:
327 ExpectationCollectionPtrT m_Expectations;
328 MockSettings m_Settings;
329
330 [[nodiscard]]
331 constexpr signature_return_type_t<SignatureT> handle_call(
332 reporting::TypeReport overloadReport,
333 std::tuple<std::reference_wrapper<std::remove_reference_t<Params>>...>&& params,
334 util::SourceLocation from) const
335 {
336 return m_Expectations->handle_call(
337 reporting::TargetReport{
338 .name = *m_Settings.name,
339 .overloadReport = std::move(overloadReport)},
340 call::info_for_signature_t<SignatureT>{
341 .args = std::move(params),
342 .fromCategory = refQualification,
343 .fromConstness = constQualification,
344 .fromSourceLocation = std::move(from),
345 .stacktrace = stacktrace::current(m_Settings.stacktraceSkip)});
346 }
347
348 template <typename... Args>
349 [[nodiscard]]
350 constexpr auto make_expectation_builder(reporting::TypeReport overloadReport, Args&&... args) const
351 {
352 return detail::make_expectation_builder(
353 m_Expectations,
354 reporting::TargetReport{.name = *m_Settings.name, .overloadReport = std::move(overloadReport)},
355 std::forward<Args>(args)...)
356 && expectation_policies::Category<refQualification>{}
357 && expectation_policies::Constness<constQualification>{};
358 }
359 };

450 requires is_overload_set_v<FirstSignature, OtherSignatures...>
451 class Mock
452 : public detail::BasicMock<FirstSignature>,
453 public detail::BasicMock<OtherSignatures>...
454 {
455 public:
456 using detail::BasicMock<FirstSignature>::operator();
457 using detail::BasicMock<FirstSignature>::expect_call;
458 using detail::BasicMock<OtherSignatures>::operator()...;
459 using detail::BasicMock<OtherSignatures>::expect_call...;

472 {
473 }
474
479 [[nodiscard]]
480 explicit Mock(MockSettings settings)
481 : Mock{
482 detail::expectation_collection_factory<
483 util::detail::unique_list_t<
484 signature_decay_t<FirstSignature>,
485 signature_decay_t<OtherSignatures>...>>::make(),
486 complete_settings(std::move(settings))}
487 {
488 }

494
498 Mock& operator=(const Mock&) = delete;
499
503 [[nodiscard]]
504 Mock(Mock&&) = default;
505
509 Mock& operator=(Mock&&) = default;
510
511 private:
512 template <typename... Collections>
513 [[nodiscard]]
514 explicit Mock(
515 std::tuple<Collections...> collections,
516 MockSettings settings) noexcept
517 : detail::BasicMock<FirstSignature>{
518 std::get<detail::expectation_collection_ptr_for<FirstSignature>>(collections),
519 settings},
520 // clang-format off
521 detail::BasicMock<OtherSignatures>{
522 std::get<detail::expectation_collection_ptr_for<OtherSignatures>>(collections),
523 settings}...
524 // clang-format on
525 {
526 }

530 {

534 }

537 }
