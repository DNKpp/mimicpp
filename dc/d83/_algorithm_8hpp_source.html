49 template <
50 std::ranges::random_access_range Target,
51 std::ranges::random_access_range Control,
52 typename Projection = std::identity,
53 std::indirect_unary_predicate<
54 std::projected<std::ranges::iterator_t<Control>, Projection>> Predicate>
55 requires std::ranges::sized_range<Target>
56 && std::ranges::sized_range<Control>
57 /*&& std::permutable<std::ranges::iterator_t<Target>>
58 && std::permutable<std::ranges::iterator_t<Control>>*/
59 [[nodiscard]]
60 constexpr std::ranges::borrowed_subrange_t<Target> partition_by(
61 Target&& targetRange,
62 Control&& controlRange,
63 Predicate predicate,
64 Projection projection = {})
65 {
66 MIMICPP_ASSERT(std::ranges::size(targetRange) == std::ranges::size(controlRange), "Size mismatch.");
67
68 auto const targetBegin = std::ranges::begin(targetRange);
69 auto const controlBegin = std::ranges::begin(controlRange);
70 auto midpoint = std::ranges::end(targetRange);
71 auto end = std::ranges::end(controlRange);
72 for (auto iter = std::ranges::find_if_not(controlRange, std::ref(predicate), std::ref(projection));
73 iter != end;
74 iter = std::ranges::find_if_not(iter, end, std::ref(predicate), std::ref(projection)))
75 {
76 auto const index = std::ranges::distance(controlBegin, iter);
77 std::ranges::iter_swap(iter, --end);
78 std::ranges::iter_swap(std::ranges::next(targetBegin, index), --midpoint);
79 }
80
81 return {std::move(midpoint), std::ranges::end(targetRange)};
82 }

96 template <std::ranges::borrowed_range Range>
97 [[nodiscard]]
98 constexpr std::ranges::borrowed_iterator_t<Range> find_closing_token(
99 Range&& str,
100 std::ranges::range_value_t<Range> const& openingToken,
101 std::ranges::range_value_t<Range> const& closingToken)
102 {
103 auto const begin = std::ranges::begin(str);
104 auto const end = std::ranges::end(str);
105 auto closingIter = std::ranges::find(str, closingToken);
106 if (closingIter == end)
107 {
108 return closingIter;
109 }
110
111 for (auto openingIter = std::ranges::find(begin, closingIter, openingToken);
112 openingIter != closingIter
113 && closingIter != end;
114 openingIter = std::ranges::find(openingIter + 1, closingIter, openingToken))
115 {
116 closingIter = std::ranges::find(closingIter + 1, end, closingToken);
117 }
118
119 return closingIter;
120 }

136 template <std::ranges::borrowed_range Range>
137 [[nodiscard]]
138 constexpr std::ranges::borrowed_subrange_t<Range> find_next_unwrapped_token(
139 Range&& str,
140 std::string_view const token,
141 std::ranges::forward_range auto&& opening,
142 std::ranges::forward_range auto&& closing)
143 {
144 using count_t = std::ranges::range_difference_t<Range>;
145 constexpr auto countAllOf = [](auto const& source, auto const& collection) {
146 count_t count{};
147 for (auto const c : collection)
148 {
149 count += std::ranges::count(source, c);
150 }
151
152 return count;
153 };
154
155 count_t openScopes{};
156 std::ranges::borrowed_subrange_t<Range> pending{str};
157 while (auto const match = std::ranges::search(pending, token))
158 {
159 // It's important to count the matched part, because the token may also be part of either the opening-
160 // or closing-collection. But count the match part with the opening collection after the test.
161 openScopes += countAllOf(std::ranges::subrange{pending.begin(), match.begin()}, opening);
162 openScopes -= countAllOf(std::ranges::subrange{pending.begin(), match.end()}, closing);
163 MIMICPP_ASSERT(0 <= openScopes, "More scopes closed than opened.");
164 if (0 == openScopes)
165 {
166 return match;
167 }
168
169 openScopes += countAllOf(match, opening);
170
171 pending = {match.end(), pending.end()};
172 }
173
174 return {std::ranges::end(str), std::ranges::end(str)};
175 }

187 template <std::ranges::forward_range Range, std::ranges::forward_range Prefix>
188 requires std::totally_ordered_with<std::ranges::range_value_t<Range>, Prefix>
189 [[nodiscard]]
190 constexpr std::ranges::borrowed_subrange_t<Range> prefix_range(Range&& range, Prefix&& prefix)
191 {
192 auto const lower = std::ranges::lower_bound(range, prefix);
193 auto const end = std::ranges::lower_bound(
194 lower,
195 std::ranges::end(range),
196 prefix,
197 [](auto const& element, auto const& p) {
198 auto const iter = std::ranges::mismatch(element, p).in2;
199 return iter == std::ranges::end(p);
200 });
201
202 return {lower, end};
203 }

217 {
218 return std::invoke(

220 [[maybe_unused]] std::index_sequence<firstIs...> const,
221 [[maybe_unused]] std::index_sequence<secondIs...> const) {
