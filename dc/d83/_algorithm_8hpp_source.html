41 template <
42 std::ranges::random_access_range Target,
43 std::ranges::random_access_range Control,
44 typename Projection = std::identity,
45 std::indirect_unary_predicate<
46 std::projected<std::ranges::iterator_t<Control>, Projection>> Predicate>
47 requires std::ranges::sized_range<Target>
48 && std::ranges::sized_range<Control>
49 /*&& std::permutable<std::ranges::iterator_t<Target>>
50 && std::permutable<std::ranges::iterator_t<Control>>*/
51 [[nodiscard]]
52 constexpr std::ranges::borrowed_subrange_t<Target> partition_by(
53 Target&& targetRange,
54 Control&& controlRange,
55 Predicate predicate,
56 Projection projection = {})
57 {
58 assert(std::ranges::size(targetRange) == std::ranges::size(controlRange) && "Size mismatch.");
59
60 auto const targetBegin = std::ranges::begin(targetRange);
61 auto const controlBegin = std::ranges::begin(controlRange);
62 auto midpoint = std::ranges::end(targetRange);
63 auto end = std::ranges::end(controlRange);
64 for (auto iter = std::ranges::find_if_not(controlRange, std::ref(predicate), std::ref(projection));
65 iter != end;
66 iter = std::ranges::find_if_not(iter, end, std::ref(predicate), std::ref(projection)))
67 {
68 auto const index = std::ranges::distance(controlBegin, iter);
69 std::ranges::iter_swap(iter, --end);
70 std::ranges::iter_swap(std::ranges::next(targetBegin, index), --midpoint);
71 }
72
73 return {std::move(midpoint), std::ranges::end(targetRange)};
74 }
75}
