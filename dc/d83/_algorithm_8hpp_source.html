44 template <
45 std::ranges::random_access_range Target,
46 std::ranges::random_access_range Control,
47 typename Projection = std::identity,
48 std::indirect_unary_predicate<
49 std::projected<std::ranges::iterator_t<Control>, Projection>> Predicate>
50 requires std::ranges::sized_range<Target>
51 && std::ranges::sized_range<Control>
52 /*&& std::permutable<std::ranges::iterator_t<Target>>
53 && std::permutable<std::ranges::iterator_t<Control>>*/
54 [[nodiscard]]
55 constexpr std::ranges::borrowed_subrange_t<Target> partition_by(
56 Target&& targetRange,
57 Control&& controlRange,
58 Predicate predicate,
59 Projection projection = {})
60 {
61 MIMICPP_ASSERT(std::ranges::size(targetRange) == std::ranges::size(controlRange), "Size mismatch.");
62
63 auto const targetBegin = std::ranges::begin(targetRange);
64 auto const controlBegin = std::ranges::begin(controlRange);
65 auto midpoint = std::ranges::end(targetRange);
66 auto end = std::ranges::end(controlRange);
67 for (auto iter = std::ranges::find_if_not(controlRange, std::ref(predicate), std::ref(projection));
68 iter != end;
69 iter = std::ranges::find_if_not(iter, end, std::ref(predicate), std::ref(projection)))
70 {
71 auto const index = std::ranges::distance(controlBegin, iter);
72 std::ranges::iter_swap(iter, --end);
73 std::ranges::iter_swap(std::ranges::next(targetBegin, index), --midpoint);
74 }
75
76 return {std::move(midpoint), std::ranges::end(targetRange)};
77 }

91 template <std::ranges::borrowed_range Range>
92 [[nodiscard]]
93 constexpr std::ranges::borrowed_iterator_t<Range> find_closing_token(
94 Range&& str,
95 std::ranges::range_value_t<Range> const& openingToken,
96 std::ranges::range_value_t<Range> const& closingToken)
97 {
98 auto const begin = std::ranges::begin(str);
99 auto const end = std::ranges::end(str);
100 auto closingIter = std::ranges::find(str, closingToken);
101 if (closingIter == end)
102 {
103 return closingIter;
104 }
105
106 for (auto openingIter = std::ranges::find(begin, closingIter, openingToken);
107 openingIter != closingIter
108 && closingIter != end;
109 openingIter = std::ranges::find(openingIter + 1, closingIter, openingToken))
110 {
111 closingIter = std::ranges::find(closingIter + 1, end, closingToken);
112 }
113
114 return closingIter;
115 }

131 template <std::ranges::borrowed_range Range>
132 [[nodiscard]]
133 constexpr std::ranges::borrowed_subrange_t<Range> find_next_unwrapped_token(
134 Range&& str,
135 std::string_view const token,
136 std::ranges::forward_range auto&& opening,
137 std::ranges::forward_range auto&& closing)
138 {
139 using count_t = std::ranges::range_difference_t<Range>;
140 constexpr auto countAllOf = [](auto const& source, auto const& collection) {
141 count_t count{};
142 for (auto const c : collection)
143 {
144 count += std::ranges::count(source, c);
145 }
146
147 return count;
148 };
149
150 count_t openScopes{};
151 std::ranges::borrowed_subrange_t<Range> pending{str};
152 while (auto const match = std::ranges::search(pending, token))
153 {
154 // It's important to count the matched part, because the token may also be part of either the opening-
155 // or closing-collection. But count the match part with the opening collection after the test.
156 openScopes += countAllOf(std::ranges::subrange{pending.begin(), match.begin()}, opening);
157 openScopes -= countAllOf(std::ranges::subrange{pending.begin(), match.end()}, closing);
158 MIMICPP_ASSERT(0 <= openScopes, "More scopes closed than opened.");
159 if (0 == openScopes)
160 {
161 return match;
162 }
163
164 openScopes += countAllOf(match, opening);
165
166 pending = {match.end(), pending.end()};
167 }
168
169 return {std::ranges::end(str), std::ranges::end(str)};
170 }
