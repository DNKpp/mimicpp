25 template <typename Arg, typename MatchesProjection = std::identity, typename DescribeProjection = printing::PrintFn>
26 struct arg_storage
27 {
28 using matches_reference = std::invoke_result_t<MatchesProjection, const Arg&>;
29 using describe_reference = std::invoke_result_t<DescribeProjection, const Arg&>;
30
31 Arg arg;
32
33 decltype(auto) as_matches_arg() const noexcept(std::is_nothrow_invocable_v<MatchesProjection, const Arg&>)
34 {
35 return std::invoke(MatchesProjection{}, arg);
36 }
37
38 decltype(auto) as_describe_arg() const noexcept(std::is_nothrow_invocable_v<DescribeProjection, const Arg&>)
39 {
40 return std::invoke(DescribeProjection{}, arg);
41 }
42 };
43
44 template <typename T>
45 struct to_arg_storage
46 {
47 using type = arg_storage<T>;
48 };
49
50 template <typename Arg, typename MatchesProjection, typename DescribeProjection>
51 struct to_arg_storage<arg_storage<Arg, MatchesProjection, DescribeProjection>>
52 {
53 using type = arg_storage<Arg, MatchesProjection, DescribeProjection>;
54 };
55
56 template <typename T>
57 using to_arg_storage_t = typename to_arg_storage<T>::type;
58}

68 template <typename Predicate, typename... AdditionalArgs>
69 requires std::is_move_constructible_v<Predicate>
70 && (... && std::is_move_constructible_v<AdditionalArgs>)
71 class PredicateMatcher
72 {
73 private:
74 using storage_t = std::tuple<detail::to_arg_storage_t<AdditionalArgs>...>;
75 template <typename T>
76 using matches_reference_t = typename detail::to_arg_storage_t<T>::matches_reference;
77
78 public:
79 [[nodiscard]]
80 explicit constexpr PredicateMatcher(
81 Predicate predicate,
82 StringViewT fmt,
83 StringViewT invertedFmt,
84 std::tuple<AdditionalArgs...> additionalArgs = {})
85 noexcept(
86 std::is_nothrow_move_constructible_v<Predicate>
87 && (... && std::is_nothrow_move_constructible_v<AdditionalArgs>))
88 : m_Predicate{std::move(predicate)},
89 m_FormatString{std::move(fmt)},
90 m_InvertedFormatString{std::move(invertedFmt)},
91 m_AdditionalArgs{std::move(additionalArgs)}
92 {
93 }
94
95 template <typename First, typename... Others>
96 requires std::predicate<
97 const Predicate&,
98 First&,
99 Others&...,
100 matches_reference_t<AdditionalArgs>...>
101 [[nodiscard]]
102 constexpr bool matches(
103 First& first,
104 Others&... others) const
105 noexcept(
106 std::is_nothrow_invocable_v<
107 const Predicate&,
108 First&,
109 Others&...,
110 matches_reference_t<AdditionalArgs>...>)
111 {
112 return std::apply(
113 [&, this](auto&... additionalArgs) {
114 return std::invoke(
115 m_Predicate,
116 first,
117 others...,
118 additionalArgs.as_matches_arg()...);
119 },
120 m_AdditionalArgs);
121 }
122
123 [[nodiscard]]
124 constexpr StringT describe() const
125 {
126 return std::apply(
127 [&, this](auto&... additionalArgs) {
128 return format::vformat(
129 m_FormatString,
130 format::make_format_args(
131 std::invoke(
132 // std::make_format_args requires lvalue-refs, so let's transform rvalue-refs to const lvalue-refs
133 [](auto&& val) noexcept -> const auto& { return val; },
134 additionalArgs.as_describe_arg())...));
135 },
136 m_AdditionalArgs);
137 }
138
139 [[nodiscard]]
140 constexpr auto operator!() const&
141 requires std::is_copy_constructible_v<Predicate>
142 && std::is_copy_constructible_v<storage_t>
143 {
144 return make_inverted(
145 m_Predicate,
146 m_InvertedFormatString,
147 m_FormatString,
148 m_AdditionalArgs);
149 }
150
151 [[nodiscard]]
152 constexpr auto operator!() &&
153 {
154 return make_inverted(
155 std::move(m_Predicate),
156 std::move(m_InvertedFormatString),
157 std::move(m_FormatString),
158 std::move(m_AdditionalArgs));
159 }
160
161 private:
162 [[no_unique_address]] Predicate m_Predicate;
163 StringViewT m_FormatString;
164 StringViewT m_InvertedFormatString;
165 storage_t m_AdditionalArgs;
166
167 template <typename Fn>
168 [[nodiscard]]
169 static constexpr auto make_inverted(
170 Fn&& fn,
171 StringViewT fmt,
172 StringViewT invertedFmt,
173 storage_t tuple)
174 {
175 using NotFnT = decltype(std::not_fn(std::forward<Fn>(fn)));
176 return PredicateMatcher<NotFnT, detail::to_arg_storage_t<AdditionalArgs>...>{
177 std::not_fn(std::forward<Fn>(fn)),
178 std::move(fmt),
179 std::move(invertedFmt),
180 std::move(tuple)};
181 }
182 };
