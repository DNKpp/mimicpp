183 template <print_iterator OutIter>
184 OutIter stringify_expectation_report_requirement_violations(
185 OutIter out,
186 std::span<std::optional<StringT> const> const descriptions,
187 StringViewT const linePrefix)
188 {
189 MIMICPP_ASSERT(!descriptions.empty(), "Zero requirements can never be violated.");
190
191 out = std::ranges::copy(linePrefix, std::move(out)).out;
192 out = format::format_to(std::move(out), "Due to Violation(s):\n");
193
194 int withoutDescription{0};
195 for (auto const& description : descriptions)
196 {
197 if (description)
198 {
199 out = std::ranges::copy(linePrefix, std::move(out)).out;
200 out = format::format_to(std::move(out), " - ");
201 out = std::ranges::copy(*description, std::move(out)).out;
202 out = format::format_to(std::move(out), "\n");
203 }
204 else
205 {
206 ++withoutDescription;
207 }
208 }
209
210 if (0 < withoutDescription)
211 {
212 out = std::ranges::copy(linePrefix, std::move(out)).out;
213 out = format::format_to(std::move(out), " - ");
214 out = format::format_to(
215 std::move(out),
216 "{} Requirement(s) failed without further description.\n",
217 withoutDescription);
218 }
219
220 return out;
221 }

223 struct unfulfilled_reason_printer
224 {
225 template <print_iterator OutIter>
226 OutIter operator()(OutIter out, state_applicable const& state) const
227 {
228 return stringify_times_state(
229 std::move(out),
230 state.count,
231 state.min,
232 state.max);
233 }
234
235 template <print_iterator OutIter>
236 OutIter operator()(OutIter out, state_inapplicable const& state) const
237 {
238 return stringify_times_state(
239 std::move(out),
240 state.count,
241 state.min,
242 state.max);
243 }
244
245 template <print_iterator OutIter>
246 OutIter operator()([[maybe_unused]] OutIter out, [[maybe_unused]] state_saturated const& state) const
247 {
248 util::unreachable();
249 }
250
251 private:
252 template <print_iterator OutIter>
253 static OutIter stringify_times_state(OutIter out, int const current, int const min, int const max)
254 {
255 const auto verbalizeValue = [](OutIter o, int const value) {
256 MIMICPP_ASSERT(0 < value, "Invalid value.");
257
258 switch (value)
259 {
260 case 1: return format::format_to(std::move(o), "once");
261 case 2: return format::format_to(std::move(o), "twice");
262 default: return format::format_to(std::move(o), "{} times", value);
263 }
264 };
265
266 MIMICPP_ASSERT(current < min, "State doesn't denote an unsatisfied state.");
267
268 out = format::format_to(std::move(out), "matching ");
269
270 if (min == max)
271 {
272 out = format::format_to(std::move(out), "exactly ");
273 out = verbalizeValue(std::move(out), min);
274 out = format::format_to(std::move(out), " ");
275 }
276 else if (max == std::numeric_limits<int>::max())
277 {
278 out = format::format_to(std::move(out), "at least ");
279 out = verbalizeValue(std::move(out), min);
280 out = format::format_to(std::move(out), " ");
281 }
282 else
283 {
284 out = format::format_to(
285 std::move(out),
286 "between {} and {} times ",
287 min,
288 max);
289 }
290
291 return format::format_to(
292 std::move(out),
293 "was expected => requires {} further match(es).\n",
294 min - current);
295 }
296 };

301 [[nodiscard]]
302 inline StringT stringify_full_match(CallReport const& call, ExpectationReport expectation)
303 {
304 MIMICPP_ASSERT(std::holds_alternative<state_applicable>(expectation.controlReport), "Report denotes inapplicable expectation.");
305
306 StringStreamT ss{};
307 ss << "Matched ";
308 detail::stringify_call_report_from(std::ostreambuf_iterator{ss}, call);
309 detail::stringify_call_report_target(std::ostreambuf_iterator{ss}, call);
310 detail::stringify_call_report_arguments(std::ostreambuf_iterator{ss}, call, "\t");
311
312 ss << "\t" << "Chose ";
313 detail::stringify_expectation_report_from(std::ostreambuf_iterator{ss}, expectation);
314
315 if (!expectation.requirementDescriptions.empty())
316 {
317 std::ranges::sort(expectation.requirementDescriptions);
318 detail::stringify_expectation_report_requirement_adherences(
319 std::ostreambuf_iterator{ss},
320 expectation.requirementDescriptions,
321 "\t");
322 }
323 else
324 {
325 ss << "\t" << "Without any Requirements.\n";
326 }
327
328 detail::stringify_stacktrace(
329 std::ostreambuf_iterator{ss},
330 call.stacktrace);
331
332 return std::move(ss).str();
333 }

336 inline StringT stringify_inapplicable_matches(CallReport const& call, std::span<ExpectationReport> expectations)
337 {
338 MIMICPP_ASSERT(!expectations.empty(), "No expectations given.");
339
340 StringStreamT ss{};
341
342 ss << "Unmatched ";
343 detail::stringify_call_report_from(std::ostreambuf_iterator{ss}, call);
344 detail::stringify_call_report_target(std::ostreambuf_iterator{ss}, call);
345 detail::stringify_call_report_arguments(std::ostreambuf_iterator{ss}, call, "\t");
346
347 ss << expectations.size() << " inapplicable but otherwise matching Expectation(s):";
348
349 for (int i{};
350 auto& expReport : expectations)
351 {
352 ss << "\n\t#" << ++i << " ";
353 detail::stringify_expectation_report_from(std::ostreambuf_iterator{ss}, expReport);
354
355 ss << "\tBecause ";
356 std::visit(
357 std::bind_front(detail::inapplicable_reason_printer{}, std::ostreambuf_iterator{ss}),
358 expReport.controlReport);
359 ss << "\n";
360
361 std::ranges::sort(expReport.requirementDescriptions);
362 detail::stringify_expectation_report_requirement_adherences(
363 std::ostreambuf_iterator{ss},
364 expReport.requirementDescriptions,
365 "\t");
366 }
367
368 detail::stringify_stacktrace(
369 std::ostreambuf_iterator{ss},
370 call.stacktrace);
371
372 return std::move(ss).str();

375 [[nodiscard]]
376 inline StringT stringify_no_matches(CallReport const& call, std::span<NoMatchReport> noMatchReports)
377 {
378 StringStreamT ss{};
379
380 ss << "Unmatched ";
381 detail::stringify_call_report_from(std::ostreambuf_iterator{ss}, call);
382 detail::stringify_call_report_target(std::ostreambuf_iterator{ss}, call);
383 detail::stringify_call_report_arguments(std::ostreambuf_iterator{ss}, call, "\t");
384
385 std::vector<NoMatchReport*> applicableReports{};
386 for (auto& noMatch : noMatchReports)
387 {
388 if (std::holds_alternative<state_applicable>(noMatch.expectationReport.controlReport))
389 {
390 applicableReports.emplace_back(&noMatch);
391 }
392 }
393
394 if (applicableReports.empty())
395 {
396 ss << "No applicable Expectations available!\n";
397 }
398 else
399 {
400 ss << applicableReports.size() << " applicable non-matching Expectation(s):";
401
402 for (int i{};
403 auto* report : applicableReports)
404 {
405 auto& [expReport, outcomes] = *report;
406
407 ss << "\n\t#" << ++i << " ";
408 detail::stringify_expectation_report_from(std::ostreambuf_iterator{ss}, expReport);
409
410 std::span const violations = util::partition_by(
411 expReport.requirementDescriptions,
412 outcomes.outcomes,
413 std::bind_front(std::equal_to{}, true));
414 MIMICPP_ASSERT(!violations.empty(), "Zero violations do not denote a no-match.");
415 std::ranges::sort(violations);
416 detail::stringify_expectation_report_requirement_violations(
417 std::ostreambuf_iterator{ss},
418 violations,
419 "\t");
420
421 std::span const adherences{expReport.requirementDescriptions.data(), violations.data()};
422 std::ranges::sort(adherences);
423 detail::stringify_expectation_report_requirement_adherences(
424 std::ostreambuf_iterator{ss},
425 adherences,
426 "\t");
427 }
428 }
429
430 detail::stringify_stacktrace(
431 std::ostreambuf_iterator{ss},
432 call.stacktrace);
433
434 return std::move(ss).str();
435 }

453 [[nodiscard]]
454 inline StringT stringify_unhandled_exception(
455 CallReport const& call,
456 ExpectationReport const& expectationReport,
457 std::exception_ptr const& exception)
458 {
459 StringStreamT ss{};
460 ss << "Unhandled Exception ";
461
462 try
463 {
464 std::rethrow_exception(exception);
465 }
466 catch (const std::exception& e)
467 {
468 ss << "with message `" << e.what() << "`\n";
469 }
470 catch (...)
471 {
472 ss << "of unknown type.\n";
473 }
474
475 ss << "\t" << "While checking ";
476 detail::stringify_expectation_report_from(std::ostreambuf_iterator{ss}, expectationReport);
477
478 ss << "For ";
479 detail::stringify_call_report_from(std::ostreambuf_iterator{ss}, call);
480 detail::stringify_call_report_target(std::ostreambuf_iterator{ss}, call);
481 detail::stringify_call_report_arguments(std::ostreambuf_iterator{ss}, call, "\t");
482
483 detail::stringify_stacktrace(
484 std::ostreambuf_iterator{ss},
485 call.stacktrace);
486
487 return std::move(ss).str();
488 }
