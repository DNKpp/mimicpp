447 template <typename Signature, typename Mock, typename... Args>
448 constexpr signature_return_type_t<Signature> indirectly_apply_mock(Mock& mock, std::tuple<Args...>&& args)
449 {
450 using mock_ref_t = std::conditional_t<
451 ValueCategory::rvalue == signature_ref_qualification_v<Signature>,
452 Mock&&,
453 Mock&>;
454 const auto forward_apply = [&]<std::size_t... indices>([[maybe_unused]] std::index_sequence<indices...> const)
455 -> signature_return_type_t<Signature> {
456 return static_cast<mock_ref_t>(mock)(
457 std::forward<Args>(std::get<indices>(args))...);
458 };
459
460 constexpr auto sequence = std::index_sequence_for<Args...>{};
461 return forward_apply(sequence);
462 };
