420 template <typename Signature>
421 constexpr auto indirectly_apply_mock = []<typename... Args>(auto& mock, Args&&... args)
422 -> signature_return_type_t<Signature> {
423 // we get the mock either as const or non-const ref.
424 // But we must respect, whether the mock shall be called as lvalue or rvalue-ref.
425
426 if constexpr (ValueCategory::rvalue == signature_ref_qualification_v<Signature>)
427 {
428 return std::move(mock)(std::forward<Args>(args)...);
429 }
430 else
431 {
432 return mock(std::forward<Args>(args)...);
433 }
434 };
435}
