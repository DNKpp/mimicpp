132 template <typename Predicate, typename... AdditionalArgs>
133 requires std::is_move_constructible_v<Predicate>
134 && (... && std::is_move_constructible_v<AdditionalArgs>)
135 class PredicateMatcher
136 {
137 public:
138 [[nodiscard]]
139 explicit constexpr PredicateMatcher(
140 Predicate predicate,
141 StringT fmt,
142 StringT invertedFmt,
143 std::tuple<AdditionalArgs...> additionalArgs = std::tuple{}
144 ) noexcept(std::is_nothrow_move_constructible_v<Predicate>
145 && (... && std::is_nothrow_move_constructible_v<AdditionalArgs>))
146 : m_Predicate{std::move(predicate)},
147 m_FormatString{std::move(fmt)},
148 m_InvertedFormatString{std::move(invertedFmt)},
149 m_AdditionalArgs{std::move(additionalArgs)}
150 {
151 }
152
153 template <typename T>
154 requires std::predicate<const Predicate&, T&, const AdditionalArgs&...>
155 [[nodiscard]]
156 constexpr bool matches(
157 T& target
158 ) const noexcept(std::is_nothrow_invocable_v<const Predicate&, T&, const AdditionalArgs&...>)
159 {
160 return std::apply(
161 [&, this](auto&... additionalArgs)
162 {
163 return std::invoke(
164 m_Predicate,
165 target,
166 additionalArgs...);
167 },
168 m_AdditionalArgs);
169 }
170
171 [[nodiscard]]
172 constexpr StringT describe() const
173 {
174 return std::apply(
175 [&, this](auto&... additionalArgs)
176 {
177 // std::make_format_args requires lvalue-refs, so let's transform rvalue-refs to const lvalue-refs
178 constexpr auto makeLvalue = [](auto&& val) noexcept -> const auto& { return val; };
179 return format::vformat(
180 m_FormatString,
181 format::make_format_args(
182 makeLvalue(mimicpp::print(additionalArgs))...));
183 },
184 m_AdditionalArgs);
185 }
186
187 [[nodiscard]]
188 constexpr auto operator !() const &
189 requires std::is_copy_constructible_v<Predicate>
190 && (... && std::is_copy_constructible_v<AdditionalArgs>)
191 {
192 return make_inverted(
193 m_Predicate,
194 m_InvertedFormatString,
195 m_FormatString,
196 std::move(m_AdditionalArgs));
197 }
198
199 [[nodiscard]]
200 constexpr auto operator !() &&
201 {
202 return make_inverted(
203 std::move(m_Predicate),
204 std::move(m_InvertedFormatString),
205 std::move(m_FormatString),
206 std::move(m_AdditionalArgs));
207 }
208
209 private:
210 [[no_unique_address]] Predicate m_Predicate;
211 StringT m_FormatString;
212 StringT m_InvertedFormatString;
213 mutable std::tuple<AdditionalArgs...> m_AdditionalArgs{};
214
215 template <typename Fn>
216 [[nodiscard]]
217 static constexpr auto make_inverted(
218 Fn&& fn,
219 StringT fmt,
220 StringT invertedFmt,
221 std::tuple<AdditionalArgs...> tuple
222 )
223 {
224 using NotFnT = decltype(std::not_fn(std::forward<Fn>(fn)));
225 return PredicateMatcher<NotFnT, AdditionalArgs...>{
226 std::not_fn(std::forward<Fn>(fn)),
227 std::move(fmt),
228 std::move(invertedFmt),
229 std::move(tuple)
230 };
231 }
232 };
