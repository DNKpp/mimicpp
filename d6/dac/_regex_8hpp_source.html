22 template <typename Char, typename CharTraits, typename Allocator, typename RegexTraits>
23 [[nodiscard]]
24 constexpr std::basic_string<Char, CharTraits, Allocator> regex_replace_all(
25 std::basic_string<Char, CharTraits, Allocator> str,
26 std::basic_regex<Char, RegexTraits> const& regex,
27 std::basic_string_view<Char, CharTraits> const fmt)
28 {
29 using match_result_t = std::match_results<typename std::basic_string<Char, CharTraits, Allocator>::const_iterator>;
30
31 // just advance by one, so that we can find matches which involves the replaced fmt
32 // like: `> > >` with regex `> >` and fmt = `>>`
33 // => `>> >` => `>>>`
34 // When fmt is empty, we need a step size of 0.
35 auto const advanceStep = static_cast<std::ptrdiff_t>(!fmt.empty());
36
37 auto processedIter = str.cbegin();
38 match_result_t match{};
39 while (std::regex_search(processedIter, str.cend(), match, regex))
40 {
41 auto const matchBegin = match[0].first;
42 auto const matchEnd = match[0].second;
43
44 auto const nextIndex = std::distance(str.cbegin(), matchBegin + advanceStep);
45 str.replace(matchBegin, matchEnd, fmt);
46 processedIter = str.cbegin() + nextIndex;
47 }
48
49 return str;
50 }
