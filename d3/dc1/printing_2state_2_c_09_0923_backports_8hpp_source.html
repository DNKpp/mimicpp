22 template <std::ranges::forward_range Range>
23 struct cxx23_backport_printer<Range>
24 {
25 template <print_iterator OutIter>
26 static OutIter print(OutIter out, auto& range)
27 {
28 out = format::format_to(std::move(out), "[");
29 auto iter = std::ranges::begin(range);
30 if (const auto end = std::ranges::end(range);
31 iter != end)
32 {
33 out = mimicpp::print(std::move(out), *iter++);
34
35 for (; iter != end; ++iter)
36 {
37 out = format::format_to(std::move(out), ", ");
38 out = mimicpp::print(std::move(out), *iter);
39 }
40 }
41
42 return format::format_to(std::move(out), "]");
43 }
44 };

46 template <std::size_t index, print_iterator OutIter>
47 OutIter print_tuple_element(OutIter out, auto& tuple)
48 {
49 if constexpr (0u != index)
50 {
51 out = format::format_to(std::move(out), ", ");
