54 template <typename T>
55 concept backend =
56 std::copyable<T>
57 && requires(backend_traits<std::remove_cvref_t<T>> traits, std::remove_cvref_t<T> const& backend) {
58 { decltype(traits)::current() } noexcept -> std::convertible_to<std::remove_cvref_t<T>>;
59 { decltype(traits)::file_name(backend) } -> std::convertible_to<std::string_view>;
60 { decltype(traits)::function_name(backend) } -> std::convertible_to<std::string_view>;
61 { decltype(traits)::line(backend) } -> std::convertible_to<std::size_t>;

167
168 template <typename... Canary, typename Backend = source_location::InstalledBackend>

173 }
174
175 [[nodiscard]]
176 friend constexpr bool operator==(SourceLocation const& lhs, SourceLocation const& rhs) noexcept
177 {
178 return lhs.line() == rhs.line()
179 && lhs.file_name() == rhs.file_name()
180 && lhs.function_name() == rhs.function_name();
