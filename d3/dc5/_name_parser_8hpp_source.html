50 constexpr void parse_function()
51 {
52 parse();
53
54 if (m_HasConversionOperator)
55 {
56 token::reduce_as_conversion_operator_function_identifier(m_TokenStack);
57 }
58 else
59 {
60 is_suffix_of<token::FunctionIdentifier>(m_TokenStack)
61 || token::try_reduce_as_function_identifier(m_TokenStack);
62 }
63
64 token::try_reduce_as_function(m_TokenStack);
65 if (!finalize<token::Function>())
66 {
67 // Well, this is a workaround to circumvent issues with lambdas on some environments.
68 // gcc produces lambdas in form `<lambda()>` which are not recognized as actual functions.
69 token::try_reduce_as_type(m_TokenStack);
70 if (!finalize<token::Type>())
71 {
72 emit_unrecognized();
73 }
74 }
75 }

140 constexpr void parse()
141 {
142 for (lexing::token next = m_Lexer.next();
143 !std::holds_alternative<lexing::end>(next.classification);
144 next = m_Lexer.next())
145 {
146 std::visit(
147 [&](auto const& tokenClass) { handle_lexer_token(next.content, tokenClass); },
148 next.classification);
149 }
150 }

152 template <token_type EndToken>
153 constexpr bool finalize()
154 {
155 if (1u == m_TokenStack.size())
156 {
157 if (auto const* const end = std::get_if<EndToken>(&m_TokenStack.back()))
158 {
159 auto& unwrapped = unwrap_visitor(m_Visitor);
160
161 unwrapped.begin();
162 std::invoke(*end, m_Visitor);
163 unwrapped.end();
164
165 return true;
166 }
167 }
168
169 return false;
170 }

192 constexpr void handle_lexer_token([[maybe_unused]] StringViewT const content, [[maybe_unused]] lexing::space const& space)
193 {
194 if (auto* const id = match_suffix<token::Identifier>(m_TokenStack))
195 {
196 // See, whether we need to merge the current builtin identifier with another one.
197 // E.g. `long long` or `unsigned int`.
198 if (id->is_builtin()
199 && merge_with_next_token())
200 {
201 auto& curContent = std::get<StringViewT>(id->content);
202 auto const [nextContent, _] = m_Lexer.next();
203 // Merge both keywords by simply treating them as contiguous content.
204 MIMICPP_ASSERT(curContent.data() + curContent.size() == content.data(), "Violated expectation.");
205 MIMICPP_ASSERT(content.data() + content.size() = nextContent.data(), "Violated expectation.");
206 curContent = StringViewT{
207 curContent.data(),
208 nextContent.data() + nextContent.size()};
209
210 return;
211 }
212
213 token::try_reduce_as_type(m_TokenStack);
214 }
215
216 // In certain cases, a space after an identifier has semantic significance.
217 // For example, consider the type names `void ()` and `foo()`:
218 // - `void ()` represents a function type returning `void`.
219 // - `foo()` represents a function named `foo`.
220 if (auto const* const nextOp = peek_if<lexing::operator_or_punctuator>();
221 nextOp
222 && util::contains(std::array{openingAngle, openingParens, openingCurly, singleQuote, backtick}, *nextOp))
223 {
224 m_TokenStack.emplace_back(token::Space{});
225 }
226 }

234 constexpr void handle_lexer_token([[maybe_unused]] StringViewT const content, lexing::keyword const& keyword)
235 {
236 if (constKeyword == keyword)
237 {
238 auto& specs = token::get_or_emplace_specs(m_TokenStack);
239 MIMICPP_ASSERT(!specs.layers.empty(), "Zero spec layers detected.");
240 auto& top = specs.layers.back();
241 MIMICPP_ASSERT(!top.isConst, "Specs is already const.");
242 top.isConst = true;
243 }
244 else if (volatileKeyword == keyword)
245 {
246 auto& specs = token::get_or_emplace_specs(m_TokenStack);
247 MIMICPP_ASSERT(!specs.layers.empty(), "Zero spec layers detected.");
248 auto& top = specs.layers.back();
249 MIMICPP_ASSERT(!top.isVolatile, "Specs is already volatile.");
250 top.isVolatile = true;
251 }
252 else if (noexceptKeyword == keyword)
253 {
254 auto& specs = token::get_or_emplace_specs(m_TokenStack);
255 MIMICPP_ASSERT(!specs.isNoexcept, "Specs already is a noexcept.");
256 specs.isNoexcept = true;
257 }
258 else if (operatorKeyword == keyword && !process_simple_operator())
259 {
260 // Conversion operators can not be part of a scope, thus they can not appear multiple times in a single type-name.
261 MIMICPP_ASSERT(!m_HasConversionOperator, "Multiple conversion operators detected.");
262
263 m_TokenStack.emplace_back(token::OperatorKeyword{});
264 m_HasConversionOperator = true;
265 }
266 else if (constexpr std::array collection{classKeyword, structKeyword, enumKeyword};
267 util::contains(collection, keyword))
268 {
269 // This token is needed, so we do not accidentally treat e.g. `(anonymous class)` as function args,
270 // because otherwise there would just be the `anonymous` identifier left.
271 m_TokenStack.emplace_back(token::TypeContext{.content = content});
272 }
273 else if (util::contains(typeKeywordCollection, keyword))
274 {
275 m_TokenStack.emplace_back(
276 token::Identifier{
277 .isBuiltinType = true,
278 .content = content});
279 }
280 }

282 constexpr bool process_simple_operator()
283 {
284 auto dropSpaceInput = [this] {
285 if (std::holds_alternative<lexing::space>(m_Lexer.peek().classification))
286 {
287 std::ignore = m_Lexer.next();
288 }
289 };
290
291 dropSpaceInput();
292
293 // As we assume valid input, we do not have to check for the actual symbol.
294 if (auto const next = m_Lexer.peek();
295 auto const* operatorToken = std::get_if<lexing::operator_or_punctuator>(&next.classification))
296 {
297 std::ignore = m_Lexer.next();
298
299 auto const finishMultiOpOperator = [&, this]([[maybe_unused]] lexing::operator_or_punctuator const& expectedClosingOp) {
300 auto const [closingContent, classification] = m_Lexer.next();
301 MIMICPP_ASSERT(lexing::token_class{expectedClosingOp} == classification, "Invalid input.");
302
303 StringViewT const content{
304 next.content.data(),
305 next.content.size() + closingContent.size()};
306 m_TokenStack.emplace_back(
307 token::Identifier{
308 .content = token::Identifier::OperatorInfo{.symbol = content}});
309 };
310
311 if (openingParens == *operatorToken)
312 {
313 finishMultiOpOperator(closingParens);
314 }
315 else if (openingSquare == *operatorToken)
316 {
317 finishMultiOpOperator(closingSquare);
318 }
319 // `operator <` and `operator <<` needs to be handled carefully, as it may come in as a template:
320 // `operator<<>` is actually `operator< <>`.
321 // Note: No tested c++ compiler actually allows `operator<<>`, but some environments still procude this.
322 else if (leftShift == *operatorToken)
323 {
324 dropSpaceInput();
325
326 if (auto const* const nextOp = peek_if<lexing::operator_or_punctuator>();
327 nextOp
328 // When next token starts a function or template, we know it's actually `operator <<`.
329 && (openingParens == *nextOp || openingAngle == *nextOp))
330 {
331 m_TokenStack.emplace_back(
332 token::Identifier{
333 .content = token::Identifier::OperatorInfo{.symbol = next.content}});
334 }
335 // looks like an `operator< <>`, so just treat both `<` separately.
336 else
337 {
338 m_TokenStack.emplace_back(
339 token::Identifier{
340 .content = token::Identifier::OperatorInfo{.symbol = next.content.substr(0u, 1u)}});
341 handle_lexer_token(next.content.substr(1u, 1u), openingAngle);
342 }
343 }
344 else
345 {
346 m_TokenStack.emplace_back(
347 token::Identifier{
348 .content = token::Identifier::OperatorInfo{.symbol = next.content}});
349 }
350
351 dropSpaceInput();
352
353 return true;
354 }
355 else if (auto const* keywordToken = std::get_if<lexing::keyword>(&next.classification);
356 keywordToken
357 && util::contains(std::array{newKeyword, deleteKeyword, coAwaitKeyword}, *keywordToken))
358 {
359 std::ignore = m_Lexer.next();
360
361 StringViewT content = next.content;
362
363 if (newKeyword == *keywordToken || deleteKeyword == *keywordToken)
364 {
365 dropSpaceInput();
366
367 if (auto const* const opAfter = peek_if<lexing::operator_or_punctuator>();
368 opAfter
369 && openingSquare == *opAfter)
370 {
371 // Strip `[]` or `[ ]` from the input.
372 std::ignore = m_Lexer.next();
373 dropSpaceInput();
374 auto const closing = m_Lexer.next();
375 MIMICPP_ASSERT(closingSquare == std::get<lexing::operator_or_punctuator>(closing.classification), "Invalid input.");
376
377 content = StringViewT{
378 next.content.data(),
379 closing.content.data() + closing.content.size()};
380 }
381 }
382
383 m_TokenStack.emplace_back(
384 token::Identifier{
385 .content = token::Identifier::OperatorInfo{.symbol = content}});
386
387 dropSpaceInput();
388
389 return true;
390 }
391
392 return false;
393 }

395 constexpr void handle_lexer_token(StringViewT const content, lexing::operator_or_punctuator const& token)
396 {
397 if (scopeResolution == token)
398 {
399 token::try_reduce_as_function_identifier(m_TokenStack);
400
401 m_TokenStack.emplace_back(
402 std::in_place_type<token::ScopeResolution>,
403 content);
404 token::try_reduce_as_scope_sequence(m_TokenStack);
405 }
406 else if (commaSeparator == token)
407 {
408 if (is_suffix_of<token::Type>(m_TokenStack)
409 || token::try_reduce_as_type(m_TokenStack))
410 {
411 token::try_reduce_as_arg_sequence(m_TokenStack);
412 }
413
414 m_TokenStack.emplace_back(
415 std::in_place_type<token::ArgSeparator>,
416 content);
417 }
418 else if (lvalueRef == token)
419 {
420 auto& specs = token::get_or_emplace_specs(m_TokenStack);
421 MIMICPP_ASSERT(token::Specs::Refness::none == specs.refness, "Specs already is a reference.");
422 specs.refness = token::Specs::Refness::lvalue;
423 }
424 else if (rvalueRef == token)
425 {
426 auto& specs = token::get_or_emplace_specs(m_TokenStack);
427 MIMICPP_ASSERT(token::Specs::Refness::none == specs.refness, "Specs already is a reference.");
428 specs.refness = token::Specs::Refness::rvalue;
429 }
430 else if (pointer == token)
431 {
432 auto& specs = token::get_or_emplace_specs(m_TokenStack);
433 specs.layers.emplace_back();
434 }
435 else if (openingAngle == token)
436 {
437 m_TokenStack.emplace_back(
438 std::in_place_type<token::OpeningAngle>,
439 content);
440 }
441 else if (closingAngle == token)
442 {
443 if (is_suffix_of<token::Type>(m_TokenStack)
444 || token::try_reduce_as_type(m_TokenStack))
445 {
446 token::try_reduce_as_arg_sequence(m_TokenStack);
447 }
448
449 m_TokenStack.emplace_back(
450 std::in_place_type<token::ClosingAngle>,
451 content);
452 token::try_reduce_as_template_identifier(m_TokenStack)
453 || token::try_reduce_as_placeholder_identifier_wrapped<token::OpeningAngle, token::ClosingAngle>(m_TokenStack);
454 }
455 else if (openingParens == token)
456 {
457 m_TokenStack.emplace_back(
458 std::in_place_type<token::OpeningParens>,
459 content);
460 }
461 else if (closingParens == token)
462 {
463 bool isNextOpeningParens{false};
464 if (auto const* const nextOp = peek_if<lexing::operator_or_punctuator>())
465 {
466 isNextOpeningParens = (openingParens == *nextOp);
467 }
468
469 // There can be no `(` directly after function-args, thus do not perform any reduction if such a token is found.
470 // This helps when function-ptrs are given, so that we do not accidentally reduce something like `(__cdecl*)` as function-args.
471 if (!isNextOpeningParens)
472 {
473 if (is_suffix_of<token::Type>(m_TokenStack)
474 || token::try_reduce_as_type(m_TokenStack))
475 {
476 token::try_reduce_as_arg_sequence(m_TokenStack);
477 }
478 }
479
480 m_TokenStack.emplace_back(
481 std::in_place_type<token::ClosingParens>,
482 content);
483
484 if (bool const result = isNextOpeningParens
485 ? token::try_reduce_as_function_ptr(m_TokenStack)
486 : token::try_reduce_as_function_context(m_TokenStack);
487 !result)
488 {
489 token::try_reduce_as_placeholder_identifier_wrapped<token::OpeningParens, token::ClosingParens>(m_TokenStack);
490 }
491 }
492 else if (openingCurly == token)
493 {
494 m_TokenStack.emplace_back(
495 std::in_place_type<token::OpeningCurly>,
496 content);
497 }
498 else if (closingCurly == token)
499 {
500 m_TokenStack.emplace_back(
501 std::in_place_type<token::ClosingCurly>,
502 content);
503 token::try_reduce_as_placeholder_identifier_wrapped<token::OpeningCurly, token::ClosingCurly>(m_TokenStack);
504 }
505 else if (backtick == token)
506 {
507 m_TokenStack.emplace_back(
508 std::in_place_type<token::OpeningBacktick>,
509 content);
510 }
511 else if (singleQuote == token)
512 {
513 if (token::try_reduce_as_function_identifier(m_TokenStack))
514 {
515 unwrap_msvc_like_function();
516 }
517 // Something like `id1::id2' should become id1::id2, so just remove the leading backtick.
518 else if (is_suffix_of<token::OpeningBacktick, token::ScopeSequence, token::Identifier>(m_TokenStack))
519 {
520 m_TokenStack.erase(m_TokenStack.cend() - 3u);
521 }
522 else
523 {
524 m_TokenStack.emplace_back(
525 std::in_place_type<token::ClosingSingleQuote>,
526 content);
527 // Well, some environments wrap in `' (like msvc) and some wrap in '' (libc++).
528 token::try_reduce_as_placeholder_identifier_wrapped<token::OpeningBacktick, token::ClosingSingleQuote>(m_TokenStack)
529 || token::try_reduce_as_placeholder_identifier_wrapped<token::ClosingSingleQuote, token::ClosingSingleQuote>(m_TokenStack);
530 }
531 }
532 // The current parsing process will never receive an `<<` or `>>` without a preceding `operator` keyword.
533 // As the current `operator` parsing currently consumes the next op-symbol, we will never reach this point
534 // with an actual left or right-shift. So, to make that easier, just split them.
535 else if (leftShift == token)
536 {
537 handle_lexer_token(content.substr(0, 1u), openingAngle);
538 handle_lexer_token(content.substr(1u, 1u), openingAngle);
539 }
540 else if (rightShift == token)
541 {
542 handle_lexer_token(content.substr(0, 1u), closingAngle);
543 handle_lexer_token(content.substr(1u, 1u), closingAngle);
544 }
545 // The msvc c++23 `std::stacktrace` implementation adds `+0x\d+` to function identifiers.
546 // The only reason to receive a `+`-token without an `operator`-token is exactly that case.
547 // So, just ignore it and skip the next identifier.
548 else if (plus == token)
549 {
550 if (auto const* const nextId = peek_if<lexing::identifier>();
551 nextId
552 && nextId->content.starts_with("0x"))
553 {
554 std::ignore = m_Lexer.next();
555 }
556 }
557 // The msvc c++23 `std::stacktrace` implementation seems to add something which looks like the executable-name as prefix.
558 // The only reason to receive a `!`-token without an `operator`-token is exactly that case.
559 // So, just ignore it and skip the previous identifier.
560 else if (exclamationMark == token
561 && is_suffix_of<token::Identifier>(m_TokenStack))
562 {
563 m_TokenStack.pop_back();
564 }
565 }
566
567 void unwrap_msvc_like_function()
568 {
569 MIMICPP_ASSERT(is_suffix_of<token::FunctionIdentifier>(m_TokenStack), "Invalid state.");
570
571 auto funIdentifier = std::get<token::FunctionIdentifier>(m_TokenStack.back());
572 m_TokenStack.pop_back();
573
574 std::optional<token::ScopeSequence> scopes{};
575 if (auto* const scopeSeq = match_suffix<token::ScopeSequence>(m_TokenStack))
576 {
577 scopes = std::move(*scopeSeq);
578 m_TokenStack.pop_back();
579 }
580
581 // Ignore return-types.
582 if (is_suffix_of<token::Type>(m_TokenStack))
583 {
584 m_TokenStack.pop_back();
585 }
586
587 MIMICPP_ASSERT(match_suffix<token::OpeningBacktick>(m_TokenStack), "Invalid state.");
588 m_TokenStack.pop_back();
589
590 // As we gather spaces in front of backticks, there may be a space here, too.
591 if (is_suffix_of<token::Space>(m_TokenStack))
592 {
593 m_TokenStack.pop_back();
594 }
595
596 MIMICPP_ASSERT(!is_suffix_of<token::ScopeSequence>(m_TokenStack), "Invlid state.");
597 if (scopes)
598 {
599 m_TokenStack.emplace_back(*std::move(scopes));
600 }
601
602 m_TokenStack.emplace_back(std::move(funIdentifier));
603 }
604 };
