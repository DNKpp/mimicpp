48 constexpr void parse_function()
49 {
50 parse();
51
52 if (m_HasConversionOperator)
53 {
54 token::reduce_as_conversion_operator_function_identifier(m_TokenStack);
55 }
56 else
57 {
58 is_suffix_of<token::FunctionIdentifier>(m_TokenStack)
59 || token::try_reduce_as_function_identifier(m_TokenStack);
60 }
61
62 token::try_reduce_as_function(m_TokenStack);
63 if (!finalize<token::Function>())
64 {
65 // Well, this is a workaround to circumvent issues with lambdas on some environments.
66 // gcc produces lambdas in form `<lambda()>` which are not recognized as actual functions.
67 token::try_reduce_as_type(m_TokenStack);
68 if (!finalize<token::Type>())
69 {
70 emit_unrecognized();
71 }
72 }
73 }

138 constexpr void parse()
139 {
140 for (lexing::token next = m_Lexer.next();
141 !std::holds_alternative<lexing::end>(next.classification);
142 next = m_Lexer.next())
143 {
144 std::visit(
145 [&](auto const& tokenClass) { handle_lexer_token(next.content, tokenClass); },
146 next.classification);
147 }
148 }

150 template <token_type EndToken>
151 constexpr bool finalize()
152 {
153 if (1u == m_TokenStack.size())
154 {
155 if (auto const* const end = std::get_if<EndToken>(&m_TokenStack.back()))
156 {
157 auto& unwrapped = unwrap_visitor(m_Visitor);
158
159 unwrapped.begin();
160 std::invoke(*end, m_Visitor);
161 unwrapped.end();
162
163 return true;
164 }
165 }
166
167 return false;
168 }

190 constexpr void handle_lexer_token([[maybe_unused]] StringViewT const content, [[maybe_unused]] lexing::space const& space)
191 {
192 if (auto* const id = match_suffix<token::Identifier>(m_TokenStack))
193 {
194 // See, whether we need to merge the current builtin identifier with another one.
195 // E.g. `long long` or `unsigned int`.
196 if (id->is_builtin()
197 && merge_with_next_token())
198 {
199 auto& curContent = std::get<StringViewT>(id->content);
200 auto const [nextContent, _] = m_Lexer.next();
201 // Merge both keywords by simply treating them as contiguous content.
202 MIMICPP_ASSERT(curContent.data() + curContent.size() == content.data(), "Violated expectation.");
203 MIMICPP_ASSERT(content.data() + content.size() = nextContent.data(), "Violated expectation.");
204 curContent = StringViewT{
205 curContent.data(),
206 nextContent.data() + nextContent.size()};
207
208 return;
209 }
210
211 token::try_reduce_as_type(m_TokenStack);
212 }
213
214 // In certain cases, a space after an identifier has semantic significance.
215 // For example, consider the type names `void ()` and `foo()`:
216 // - `void ()` represents a function type returning `void`.
217 // - `foo()` represents a function named `foo`.
218 if (auto const* const nextOp = peek_if<lexing::operator_or_punctuator>();
219 nextOp
220 && util::contains(std::array{openingAngle, openingParens, openingCurly, singleQuote, backtick}, *nextOp))
221 {
222 m_TokenStack.emplace_back(token::Space{});
223 }
224 }

232 constexpr void handle_lexer_token([[maybe_unused]] StringViewT const content, lexing::keyword const& keyword)
233 {
234 if (constKeyword == keyword)
235 {
236 auto& specs = token::get_or_emplace_specs(m_TokenStack);
237 MIMICPP_ASSERT(!specs.layers.empty(), "Zero spec layers detected.");
238 auto& top = specs.layers.back();
239 MIMICPP_ASSERT(!top.isConst, "Specs is already const.");
240 top.isConst = true;
241 }
242 else if (volatileKeyword == keyword)
243 {
244 auto& specs = token::get_or_emplace_specs(m_TokenStack);
245 MIMICPP_ASSERT(!specs.layers.empty(), "Zero spec layers detected.");
246 auto& top = specs.layers.back();
247 MIMICPP_ASSERT(!top.isVolatile, "Specs is already volatile.");
248 top.isVolatile = true;
249 }
250 else if (noexceptKeyword == keyword)
251 {
252 auto& specs = token::get_or_emplace_specs(m_TokenStack);
253 MIMICPP_ASSERT(!specs.isNoexcept, "Specs already is a noexcept.");
254 specs.isNoexcept = true;
255 }
256 else if (operatorKeyword == keyword && !process_simple_operator())
257 {
258 // Conversion operators can not be part of a scope, thus they can not appear multiple times in a single type-name.
259 MIMICPP_ASSERT(!m_HasConversionOperator, "Multiple conversion operators detected.");
260
261 m_TokenStack.emplace_back(token::OperatorKeyword{});
262 m_HasConversionOperator = true;
263 }
264 else if (constexpr std::array collection{classKeyword, structKeyword, enumKeyword};
265 util::contains(collection, keyword))
266 {
267 // This token is needed, so we do not accidentally treat e.g. `(anonymous class)` as function args,
268 // because otherwise there would just be the `anonymous` identifier left.
269 m_TokenStack.emplace_back(token::TypeContext{.content = content});
270 }
271 else if (util::contains(typeKeywordCollection, keyword))
272 {
273 m_TokenStack.emplace_back(
274 token::Identifier{
275 .isBuiltinType = true,
276 .content = content});
277 }
278 }

280 constexpr bool process_simple_operator()
281 {
282 auto dropSpaceInput = [this] {
283 if (std::holds_alternative<lexing::space>(m_Lexer.peek().classification))
284 {
285 std::ignore = m_Lexer.next();
286 }
287 };
288
289 dropSpaceInput();
290
291 // As we assume valid input, we do not have to check for the actual symbol.
292 if (auto const next = m_Lexer.peek();
293 auto const* operatorToken = std::get_if<lexing::operator_or_punctuator>(&next.classification))
294 {
295 std::ignore = m_Lexer.next();
296
297 auto const finishMultiOpOperator = [&, this](lexing::operator_or_punctuator const& expectedClosingOp) {
298 auto const [closingContent, classification] = m_Lexer.next();
299 MIMICPP_ASSERT(lexing::token_class{expectedClosingOp} == classification, "Invalid input.");
300
301 StringViewT const content{
302 next.content.data(),
303 next.content.size() + closingContent.size()};
304 m_TokenStack.emplace_back(
305 token::Identifier{
306 .content = token::Identifier::OperatorInfo{.symbol = content}});
307 };
308
309 if (openingParens == *operatorToken)
310 {
311 finishMultiOpOperator(closingParens);
312 }
313 else if (openingSquare == *operatorToken)
314 {
315 finishMultiOpOperator(closingSquare);
316 }
317 // `operator <` and `operator <<` needs to be handled carefully, as it may come in as a template:
318 // `operator<<>` is actually `operator< <>`.
319 // Note: No tested c++ compiler actually allows `operator<<>`, but some environments still procude this.
320 else if (leftShift == *operatorToken)
321 {
322 dropSpaceInput();
323
324 if (auto const* const nextOp = peek_if<lexing::operator_or_punctuator>();
325 nextOp
326 // When next token starts a function or template, we know it's actually `operator <<`.
327 && (openingParens == *nextOp || openingAngle == *nextOp))
328 {
329 m_TokenStack.emplace_back(
330 token::Identifier{
331 .content = token::Identifier::OperatorInfo{.symbol = next.content}});
332 }
333 // looks like an `operator< <>`, so just treat both `<` separately.
334 else
335 {
336 m_TokenStack.emplace_back(
337 token::Identifier{
338 .content = token::Identifier::OperatorInfo{.symbol = next.content.substr(0u, 1u)}});
339 handle_lexer_token(next.content.substr(1u, 1u), openingAngle);
340 }
341 }
342 else
343 {
344 m_TokenStack.emplace_back(
345 token::Identifier{
346 .content = token::Identifier::OperatorInfo{.symbol = next.content}});
347 }
348
349 dropSpaceInput();
350
351 return true;
352 }
353 else if (auto const* keywordToken = std::get_if<lexing::keyword>(&next.classification);
354 keywordToken
355 && util::contains(std::array{newKeyword, deleteKeyword, coAwaitKeyword}, *keywordToken))
356 {
357 std::ignore = m_Lexer.next();
358
359 StringViewT content = next.content;
360
361 if (newKeyword == *keywordToken || deleteKeyword == *keywordToken)
362 {
363 dropSpaceInput();
364
365 if (auto const* const opAfter = peek_if<lexing::operator_or_punctuator>();
366 opAfter
367 && openingSquare == *opAfter)
368 {
369 // Strip `[]` or `[ ]` from the input.
370 std::ignore = m_Lexer.next();
371 dropSpaceInput();
372 auto const closing = m_Lexer.next();
373 MIMICPP_ASSERT(closingSquare == std::get<lexing::operator_or_punctuator>(closing.classification), "Invalid input.");
374
375 content = StringViewT{
376 next.content.data(),
377 closing.content.data() + closing.content.size()};
378 }
379 }
380
381 m_TokenStack.emplace_back(
382 token::Identifier{
383 .content = token::Identifier::OperatorInfo{.symbol = content}});
384
385 dropSpaceInput();
386
387 return true;
388 }
389
390 return false;
391 }

393 constexpr void handle_lexer_token(StringViewT const content, lexing::operator_or_punctuator const& token)
394 {
395 if (scopeResolution == token)
396 {
397 token::try_reduce_as_function_identifier(m_TokenStack);
398
399 m_TokenStack.emplace_back(
400 std::in_place_type<token::ScopeResolution>,
401 content);
402 token::try_reduce_as_scope_sequence(m_TokenStack);
403 }
404 else if (commaSeparator == token)
405 {
406 if (is_suffix_of<token::Type>(m_TokenStack)
407 || token::try_reduce_as_type(m_TokenStack))
408 {
409 token::try_reduce_as_arg_sequence(m_TokenStack);
410 }
411
412 m_TokenStack.emplace_back(
413 std::in_place_type<token::ArgSeparator>,
414 content);
415 }
416 else if (lvalueRef == token)
417 {
418 auto& specs = token::get_or_emplace_specs(m_TokenStack);
419 MIMICPP_ASSERT(token::Specs::Refness::none == specs.refness, "Specs already is a reference.");
420 specs.refness = token::Specs::Refness::lvalue;
421 }
422 else if (rvalueRef == token)
423 {
424 auto& specs = token::get_or_emplace_specs(m_TokenStack);
425 MIMICPP_ASSERT(token::Specs::Refness::none == specs.refness, "Specs already is a reference.");
426 specs.refness = token::Specs::Refness::rvalue;
427 }
428 else if (pointer == token)
429 {
430 auto& specs = token::get_or_emplace_specs(m_TokenStack);
431 specs.layers.emplace_back();
432 }
433 else if (openingAngle == token)
434 {
435 m_TokenStack.emplace_back(
436 std::in_place_type<token::OpeningAngle>,
437 content);
438 }
439 else if (closingAngle == token)
440 {
441 if (is_suffix_of<token::Type>(m_TokenStack)
442 || token::try_reduce_as_type(m_TokenStack))
443 {
444 token::try_reduce_as_arg_sequence(m_TokenStack);
445 }
446
447 m_TokenStack.emplace_back(
448 std::in_place_type<token::ClosingAngle>,
449 content);
450 token::try_reduce_as_template_identifier(m_TokenStack)
451 || token::try_reduce_as_placeholder_identifier_wrapped<token::OpeningAngle, token::ClosingAngle>(m_TokenStack);
452 }
453 else if (openingParens == token)
454 {
455 m_TokenStack.emplace_back(
456 std::in_place_type<token::OpeningParens>,
457 content);
458 }
459 else if (closingParens == token)
460 {
461 bool isNextOpeningParens{false};
462 if (auto const* const nextOp = peek_if<lexing::operator_or_punctuator>())
463 {
464 isNextOpeningParens = (openingParens == *nextOp);
465 }
466
467 // There can be no `(` directly after function-args, thus do not perform any reduction if such a token is found.
468 // This helps when function-ptrs are given, so that we do not accidentally reduce something like `(__cdecl*)` as function-args.
469 if (!isNextOpeningParens)
470 {
471 if (is_suffix_of<token::Type>(m_TokenStack)
472 || token::try_reduce_as_type(m_TokenStack))
473 {
474 token::try_reduce_as_arg_sequence(m_TokenStack);
475 }
476 }
477
478 m_TokenStack.emplace_back(
479 std::in_place_type<token::ClosingParens>,
480 content);
481
482 if (bool const result = isNextOpeningParens
483 ? token::try_reduce_as_function_ptr(m_TokenStack)
484 : token::try_reduce_as_function_context(m_TokenStack);
485 !result)
486 {
487 token::try_reduce_as_placeholder_identifier_wrapped<token::OpeningParens, token::ClosingParens>(m_TokenStack);
488 }
489 }
490 else if (openingCurly == token)
491 {
492 m_TokenStack.emplace_back(
493 std::in_place_type<token::OpeningCurly>,
494 content);
495 }
496 else if (closingCurly == token)
497 {
498 m_TokenStack.emplace_back(
499 std::in_place_type<token::ClosingCurly>,
500 content);
501 token::try_reduce_as_placeholder_identifier_wrapped<token::OpeningCurly, token::ClosingCurly>(m_TokenStack);
502 }
503 else if (backtick == token)
504 {
505 m_TokenStack.emplace_back(
506 std::in_place_type<token::OpeningBacktick>,
507 content);
508 }
509 else if (singleQuote == token)
510 {
511 if (token::try_reduce_as_function_identifier(m_TokenStack))
512 {
513 unwrap_msvc_like_function();
514 }
515 // Something like `id1::id2' should become id1::id2, so just remove the leading backtick.
516 else if (is_suffix_of<token::OpeningBacktick, token::ScopeSequence, token::Identifier>(m_TokenStack))
517 {
518 m_TokenStack.erase(m_TokenStack.cend() - 3u);
519 }
520 else
521 {
522 m_TokenStack.emplace_back(
523 std::in_place_type<token::ClosingSingleQuote>,
524 content);
525 // Well, some environments wrap in `' (like msvc) and some wrap in '' (libc++).
526 token::try_reduce_as_placeholder_identifier_wrapped<token::OpeningBacktick, token::ClosingSingleQuote>(m_TokenStack)
527 || token::try_reduce_as_placeholder_identifier_wrapped<token::ClosingSingleQuote, token::ClosingSingleQuote>(m_TokenStack);
528 }
529 }
530 // The current parsing process will never receive an `<<` or `>>` without a preceding `operator` keyword.
531 // As the current `operator` parsing currently consumes the next op-symbol, we will never reach this point
532 // with an actual left or right-shift. So, to make that easier, just split them.
533 else if (leftShift == token)
534 {
535 handle_lexer_token(content.substr(0, 1u), openingAngle);
536 handle_lexer_token(content.substr(1u, 1u), openingAngle);
537 }
538 else if (rightShift == token)
539 {
540 handle_lexer_token(content.substr(0, 1u), closingAngle);
541 handle_lexer_token(content.substr(1u, 1u), closingAngle);
542 }
543 // The msvc c++23 `std::stacktrace` implementation adds `+0x\d+` to function identifiers.
544 // The only reason to receive a `+`-token without an `operator`-token is exactly that case.
545 // So, just ignore it and skip the next identifier.
546 else if (plus == token)
547 {
548 if (auto const* const nextId = peek_if<lexing::identifier>();
549 nextId
550 && nextId->content.starts_with("0x"))
551 {
552 std::ignore = m_Lexer.next();
553 }
554 }
555 // The msvc c++23 `std::stacktrace` implementation seems to add something which looks like the executable-name as prefix.
556 // The only reason to receive a `!`-token without an `operator`-token is exactly that case.
557 // So, just ignore it and skip the previous identifier.
558 else if (exclamationMark == token
559 && is_suffix_of<token::Identifier>(m_TokenStack))
560 {
561 m_TokenStack.pop_back();
562 }
563 }
564
565 void unwrap_msvc_like_function()
566 {
567 MIMICPP_ASSERT(is_suffix_of<token::FunctionIdentifier>(m_TokenStack), "Invalid state.");
568
569 auto funIdentifier = std::get<token::FunctionIdentifier>(m_TokenStack.back());
570 m_TokenStack.pop_back();
571
572 std::optional<token::ScopeSequence> scopes{};
573 if (auto* const scopeSeq = match_suffix<token::ScopeSequence>(m_TokenStack))
574 {
575 scopes = std::move(*scopeSeq);
576 m_TokenStack.pop_back();
577 }
578
579 // Ignore return-types.
580 if (is_suffix_of<token::Type>(m_TokenStack))
581 {
582 m_TokenStack.pop_back();
583 }
584
585 MIMICPP_ASSERT(match_suffix<token::OpeningBacktick>(m_TokenStack), "Invalid state.");
586 m_TokenStack.pop_back();
587
588 // As we gather spaces in front of backticks, there may be a space here, too.
589 if (is_suffix_of<token::Space>(m_TokenStack))
590 {
591 m_TokenStack.pop_back();
592 }
593
594 MIMICPP_ASSERT(!is_suffix_of<token::ScopeSequence>(m_TokenStack), "Invlid state.");
595 if (scopes)
596 {
597 m_TokenStack.emplace_back(*std::move(scopes));
598 }
599
600 m_TokenStack.emplace_back(std::move(funIdentifier));
601 }
602 };
