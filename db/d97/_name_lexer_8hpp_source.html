199 class NameLexer
200 {
201 public:
202 [[nodiscard]]
203 explicit constexpr NameLexer(StringViewT text) noexcept
204 : m_Text{std::move(text)},
205 m_Next{find_next()}

207 }

209 [[nodiscard]]
210 constexpr token next() noexcept
211 {
212 return std::exchange(m_Next, find_next());
213 }
214
215 [[nodiscard]]
216 constexpr token const& peek() const noexcept
217 {
218 return m_Next;
219 }
220
221 private:
222 StringViewT m_Text;
223 token m_Next;
224
225 [[nodiscard]]
226 constexpr token find_next() noexcept
227 {
228 if (m_Text.empty())
229 {
230 return token{
231 .content = {m_Text.cend(), m_Text.cend()},
232 .classification = end{}
233 };
234 }
235
236 if (is_space(m_Text.front()))
237 {
238 // Multiple consecutive spaces or any whitespace character other than a single space
239 // carry no meaningful semantic value beyond delimitation.
240 // Although single spaces may sometimes influence the result and sometimes not,
241 // complicating the overall process, we filter out all non-single whitespace characters here.
242 if (StringViewT const content = next_as_space();
243 " " == content)
244 {
245 return token{
246 .content = content,
247 .classification = space{}};
248 }
249
250 return find_next();
251 }
252
253 if (auto const options = util::prefix_range(
254 operator_or_punctuator::textCollection,
255 m_Text.substr(0u, 1u)))
256 {
257 return next_as_op_or_punctuator(options);
258 }
259
260 StringViewT const content = next_as_identifier();
261 // As we do not perform any prefix-checks, we need to check now whether the token actually denotes a keyword.
262 if (auto const iter = util::binary_find(keyword::textCollection, content);
263 iter != keyword::textCollection.cend())
264 {
265 return token{
266 .content = content,
267 .classification = keyword{std::ranges::distance(keyword::textCollection.begin(), iter)}};
268 }
269
270 return token{
271 .content = content,
272 .classification = identifier{.content = content}};
273 }
274
275 [[nodiscard]]
276 constexpr StringViewT next_as_space() noexcept
277 {
278 auto const end = std::ranges::find_if_not(m_Text.cbegin() + 1, m_Text.cend(), is_space);
279 StringViewT const content{m_Text.cbegin(), end};
280 m_Text = StringViewT{end, m_Text.cend()};
281
282 return content;
283 }
284
289 [[nodiscard]]
290 constexpr token next_as_op_or_punctuator(std::span<StringViewT const> options) noexcept
291 {
292 MIMICPP_ASSERT(m_Text.substr(0u, 1u) == options.front(), "Assumption does not hold.");
293
294 auto const try_advance = [&, this](std::size_t const n) {
295 if (n <= m_Text.size())
296 {
297 return util::prefix_range(
298 options,
299 StringViewT{m_Text.cbegin(), m_Text.cbegin() + n});
300 }
301
302 return std::ranges::subrange{options.end(), options.end()};
303 };
304
305 std::size_t length{1u};
306 StringViewT const* lastMatch = &options.front();
307 while (auto const nextOptions = try_advance(length + 1))
308 {
309 ++length;
310 options = {nextOptions.begin(), nextOptions.end()};
311
312 // If the first string is exactly the size of the prefix, it's a match.
313 if (auto const& front = options.front();
314 length == front.size())
315 {
316 lastMatch = &front;
317 }
318 }
319
320 MIMICPP_ASSERT(!options.empty(), "Invalid state.");
321 MIMICPP_ASSERT(lastMatch, "Invalid state.");
322
323 auto const index = std::ranges::distance(operator_or_punctuator::textCollection.data(), lastMatch);
324 StringViewT const content{m_Text.substr(0u, lastMatch->size())};
325 m_Text.remove_prefix(lastMatch->size());
326
327 return token{
328 .content = content,
329 .classification = operator_or_punctuator{index}};
330 }
331
342 [[nodiscard]]
343 constexpr StringViewT next_as_identifier() noexcept
344 {
345 auto const last = std::ranges::find_if_not(
346 m_Text.cbegin() + 1,
347 m_Text.cend(),
348 [](auto const c) {
349 return !is_space(c)
350 && !std::ranges::binary_search(operator_or_punctuator::textCollection, StringViewT{&c, 1u});
351 });
352
353 StringViewT const content{m_Text.cbegin(), last};
354 m_Text = {last, m_Text.cend()};
355
356 return content;
357 }
358 };
359}
