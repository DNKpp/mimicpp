28 template <typename... Sequences>
29 [[nodiscard]]
30 constexpr std::tuple<std::tuple<std::shared_ptr<Sequences>, sequence::Id>...> make_sequence_entries(
31 const std::tuple<std::shared_ptr<Sequences>...>& sequences) noexcept
32 {
33 // This is a workaround due to some issues with clang-17 with c++23 and libstdc++
34 // That configuration prevents the direct initialization, thus we have to default construct first and
35 // setup afterwards. Compilers will probably detect that and optimize accordingly.
36 std::tuple<std::tuple<std::shared_ptr<Sequences>, sequence::Id>...> result{};
37 std::invoke(
38 [&]<std::size_t... indices>([[maybe_unused]] const std::index_sequence<indices...>) noexcept {
39 ((std::get<indices>(result) =
40 std::tuple{
41 std::get<indices>(sequences),
42 std::get<indices>(sequences)->add(),
43 }),
44 ...);
45 },
46 std::index_sequence_for<Sequences...>{});
47 return result;
48 }

93 [[nodiscard]]
94 reporting::control_state_t make_control_state(
95 const int min,
96 const int max,
97 const int count,
98 const auto& sequenceEntries)
99 {
100 if (count == max)
101 {
102 return reporting::state_saturated{
103 .min = min,
104 .max = max,
105 .count = count,
106 .sequences = std::apply(
107 [](const auto&... entries) {
108 return std::vector<sequence::Tag>{
109 std::get<0>(entries)->tag()...};
110 },
111 sequenceEntries)};
112 }
113
114 std::vector<sequence::Tag> inapplicable{};
115 std::vector<sequence::rating> ratings{};
116 std::apply(
117 [&](const auto&... entries) {
118 (...,
119 std::invoke(
120 [&](auto& seq, const sequence::Id id) {
121 if (const std::optional priority = seq->priority_of(id))
122 {
123 ratings.emplace_back(
124 *priority,
125 seq->tag());
126 }
127 else
128 {
129 inapplicable.emplace_back(seq->tag());
130 }
131 },
132 std::get<0>(entries),
133 std::get<1>(entries)));
134 },
135 sequenceEntries);
136
137 if (!std::ranges::empty(inapplicable))
138 {
139 return reporting::state_inapplicable{
140 .min = min,
141 .max = max,
142 .count = count,
143 .sequenceRatings = std::move(ratings),
144 .inapplicableSequences = std::move(inapplicable)};
145 }
146
147 return reporting::state_applicable{
148 .min = min,
149 .max = max,
150 .count = count,
151 .sequenceRatings = std::move(ratings),
152 };
153 }
154 }

156 template <typename... Sequences>
157 class ControlPolicy
158 {
159 public:
160 static constexpr std::size_t sequenceCount{sizeof...(Sequences)};
161
162 [[nodiscard]]
163 explicit constexpr ControlPolicy(
164 const detail::TimesConfig& timesConfig,
165 const sequence::detail::Config<Sequences...>& sequenceConfig) noexcept
166 : m_Min{timesConfig.min()},
167 m_Max{timesConfig.max()},
168 m_Sequences{
169 detail::make_sequence_entries(sequenceConfig.sequences())}
170 {
171 update_sequence_states();
172 }
173
174 [[nodiscard]]
175 constexpr bool is_satisfied() const noexcept
176 {
177 return m_Min <= m_Count
178 && m_Count <= m_Max;
179 }
180
181 [[nodiscard]]
182 constexpr bool is_saturated() const noexcept
183 {
184 return m_Count == m_Max;
185 }
186
187 [[nodiscard]]
188 constexpr bool is_applicable() const noexcept
189 {
190 return m_Count < m_Max
191 && std::apply(
192 [](const auto&... entries) noexcept {
193 return (... && std::get<0>(entries)->is_consumable(std::get<1>(entries)));
194 },
195 m_Sequences);
196 }
197
198 constexpr void consume() noexcept
199 {
200 MIMICPP_ASSERT(is_applicable(), "Policy is inapplicable.");
201
202 std::apply(
203 [](auto&... entries) noexcept {
204 (..., std::get<0>(entries)->consume(std::get<1>(entries)));
205 },
206 m_Sequences);
207
208 ++m_Count;
209
210 update_sequence_states();
211 }
212
213 [[nodiscard]]
214 reporting::control_state_t state() const
215 {
216 return detail::make_control_state(
217 m_Min,
218 m_Max,
219 m_Count,
220 m_Sequences);
221 }
222
223 private:
224 int m_Min;
225 int m_Max;
226 int m_Count{};
227 std::tuple<
228 std::tuple<std::shared_ptr<Sequences>, sequence::Id>...>
229 m_Sequences{};
230
231 constexpr void update_sequence_states() noexcept
232 {
233 if (m_Count == m_Min)
234 {
235 std::apply(
236 [](auto&... entries) noexcept {
237 (..., std::get<0>(entries)->set_satisfied(std::get<1>(entries)));
238 },
239 m_Sequences);
240 }
241 else if (m_Count == m_Max)
242 {
243 std::apply(
244 [](auto&... entries) noexcept {
245 (..., std::get<0>(entries)->set_saturated(std::get<1>(entries)));
246 },
247 m_Sequences);
248 }
249 }
250 };
