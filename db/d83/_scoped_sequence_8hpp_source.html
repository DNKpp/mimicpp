26 public:
27 ExpectationBuilderFinalizer(ExpectationBuilderFinalizer const&) = delete;
28 ExpectationBuilderFinalizer& operator=(ExpectationBuilderFinalizer const&) = delete;
29 ExpectationBuilderFinalizer(ExpectationBuilderFinalizer&&) = delete;

31
32 ~ExpectationBuilderFinalizer() = default;
33
34 template <bool timesConfigured, typename... Args>
35 [[nodiscard]]
36 explicit(false) constexpr ExpectationBuilderFinalizer(
37 BasicExpectationBuilder<timesConfigured, Args...>&& builder,
38 util::SourceLocation loc = {})
39 : m_Builder{&builder},
40 m_SourceLocation{std::move(loc)},
41 m_FinalizeFn{
42 +[](void* storage, util::SourceLocation finalLoc, Sequence& sequence) {
43 auto* builderPtr = static_cast<BasicExpectationBuilder<timesConfigured, Args...>*>(storage);
44 return ScopedExpectation{
45 std::move(*builderPtr) and expect::in_sequence(sequence),
46 std::move(finalLoc)};
47 }}
48 {
49 }
50
51 [[nodiscard]]
52 ScopedExpectation finalize(Sequence& sequence)
53 {
54 MIMICPP_ASSERT(m_Builder, "Builder is nullptr.");
55 MIMICPP_ASSERT(m_FinalizeFn, "FinalizeFn is nullptr.");
56
57 return std::invoke(
58 std::exchange(m_FinalizeFn, nullptr),
59 std::exchange(m_Builder, nullptr),
60 std::move(m_SourceLocation),
61 sequence);
62 }
63
64 private:
65 void* m_Builder;
66 util::SourceLocation m_SourceLocation;
67
68 using FinalizeFn = ScopedExpectation (*)(void*, util::SourceLocation, Sequence&);
69 FinalizeFn m_FinalizeFn;
70 };

112 ~BasicScopedSequence() noexcept(false)
113 {
114 auto expectations = std::exchange(m_Expectations, {});
115 while (!expectations.empty())
116 {
117 // Prevent the `pop_front() noexcept` from raising an exception and thus terminating.
118 auto const expectation = std::move(expectations.front());
119 expectations.pop_front();
120 }
121 }
122
126 [[nodiscard]]
127 BasicScopedSequence() = default;

149 BasicScopedSequence& operator+=(sequence::detail::ExpectationBuilderFinalizer<BasicScopedSequence>&& builder)
150 {
151 m_Expectations.emplace_back(builder.finalize(*this));
152
153 return *this;
154 }

162 [[nodiscard]]

165 return m_Expectations;
166 }
