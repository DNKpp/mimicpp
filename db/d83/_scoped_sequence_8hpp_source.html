27 {
28 public:
29 ExpectationBuilderFinalizer(ExpectationBuilderFinalizer const&) = delete;
30 ExpectationBuilderFinalizer& operator=(ExpectationBuilderFinalizer const&) = delete;
31 ExpectationBuilderFinalizer(ExpectationBuilderFinalizer&&) = delete;
32 ExpectationBuilderFinalizer& operator=(ExpectationBuilderFinalizer&&) = delete;
33
34 ~ExpectationBuilderFinalizer() = default;
35
36 template <bool timesConfigured, typename... Args>
37 [[nodiscard]]
38 explicit(false) ExpectationBuilderFinalizer(
39 expectation::BasicBuilder<timesConfigured, Args...>&& builder,
40 util::SourceLocation loc = {})
41 : m_FinalizeFn{
42 [&builder, loc = std::move(loc)](Sequence& sequence) mutable {
43 return ScopedExpectation{
44 std::move(builder) and expect::in_sequence(sequence),
45 std::move(loc)};
46 }}
47 {
48 }
49
50 [[nodiscard]]
51 ScopedExpectation finalize(Sequence& sequence)
52 {
53 MIMICPP_ASSERT(m_FinalizeFn, "Finalize-Function was already consumed.");
54
55 return std::invoke(
56 std::exchange(m_FinalizeFn, nullptr),
57 sequence);
58 }
59
60 private:
61 std::function<ScopedExpectation(Sequence&)> m_FinalizeFn;
62 };
