75 template <print_iterator OutIter>
76 [[nodiscard]]
77 std::tuple<OutIter, std::size_t> prettify_function_scope(
78 OutIter out,
79 auto const& prefixMatches,
80 auto const& suffixMatches)
81 {
82 MIMICPP_ASSERT(prefixMatches.size() == 2, "Regex out-of-sync.");
83 MIMICPP_ASSERT(suffixMatches.size() == 1, "Regex out-of-sync.");
84
85 StringViewT const functionName{prefixMatches[1].first, prefixMatches[1].second};
86 StringViewT const prefix{prefixMatches[0].first, prefixMatches[0].second};
87 StringViewT const suffix{suffixMatches[0].first, suffixMatches[0].second};
88
89 return {
90 format::format_to(std::move(out), "({})::", functionName),
91 static_cast<std::size_t>(std::ranges::distance(prefix.data(), suffix.data() + suffix.size()))};
92 }

94 template <print_iterator OutIter>
95 [[nodiscard]]
96 OutIter prettify_template_scope(
97 OutIter out,
98 StringViewT const scope,
99 StringViewT const templateScopeSuffix)
100 {
101 auto reversedName = scope
102 | std::views::reverse
103 | std::views::drop(templateScopeSuffix.size());
104 auto const iter = util::find_closing_token(reversedName, '>', '<');
105 MIMICPP_ASSERT(iter != reversedName.end(), "No template begin found.");
106 out = format::format_to(
107 std::move(out),
108 "{}::",
109 StringViewT{scope.cbegin(), iter.base() - 1});
110
111 return out;
112 }

120 [[nodiscard]]
121 std::tuple<OutIter, std::size_t> prettify_lambda_scope(
122 OutIter out,
123 RegexT const& suffixRegex,
124 StringViewT const prefix,
125 StringViewT const scope)
126 {
127 MIMICPP_ASSERT(scope.data() == prefix.data(), "Prefix and scope must be aligned.");
128
129 StringViewT rest{prefix.data() + prefix.size(), scope.data() + scope.size()};
130 auto const closingIter = util::find_closing_token(rest, '(', ')');
131 MIMICPP_ASSERT(closingIter != rest.cend(), "No corresponding closing-token found.");
132
133 SVMatchT suffixMatches{};
134 rest = StringViewT{closingIter, rest.cend()};
135 std::regex_search(rest.cbegin(), rest.cend(), suffixMatches, suffixRegex);
136 MIMICPP_ASSERT(!suffixMatches.empty(), "No function suffix found.");
137 MIMICPP_ASSERT(suffixMatches.size() == 2, "Regex out-of-sync.");
138 StringViewT const suffix{suffixMatches[0].first, suffixMatches[0].second};
139
140 out = std::ranges::copy(StringViewT{"lambda#"}, std::move(out)).out;
141
142 StringViewT const lambdaNo = std::invoke([&] {
143 #if MIMICPP_DETAIL_USES_LIBCXX
144 // no number set?, just default to 1
145 if (0 == suffixMatches[1].length())
146 {
147 return StringViewT{"1"};
148 }
149 #endif
150
151 return StringViewT{suffixMatches[1].first, suffixMatches[1].second};
152 });
153 out = std::ranges::copy(lambdaNo, std::move(out)).out;
154
155 return {
156 std::ranges::copy(StringViewT{"::"}, std::move(out)).out,
157 static_cast<std::size_t>(std::ranges::distance(prefix.data(), suffix.data() + suffix.size()))};
158 }

160 template <print_iterator OutIter>
161 std::tuple<OutIter, std::size_t, std::size_t> consume_next_scope(OutIter out, StringViewT const scope)
162 {
163 SVMatchT matches{};
164
165 // the anonymous namespace was already prettified, so just print it as-is
166 if (scope == anonymousNamespaceTargetScopeText)
167 {
168 return std::tuple{
169 std::ranges::copy(scope, std::move(out)).out,
170 0,
171 scope.size()};
172 }
173
174 if (constexpr StringViewT templateScopeSuffix{">::"};
175 scope.ends_with(templateScopeSuffix))
176 {
177 return std::tuple{
178 prettify_template_scope(std::move(out), scope, templateScopeSuffix),
179 0u,
180 scope.size()};
181 }
182
183 // libc++ sometimes also uses `'lambda'()` for lambdas
184 #if MIMICPP_DETAIL_USES_LIBCXX
185 if (constexpr StringViewT libcxxLambdaScopePrefix{"'lambda'("};
186 scope.starts_with(libcxxLambdaScopePrefix))
187 {
188 // use a fake group for the lambda number
189 static RegexT lambdaEndRegex{R"(\)()::$)"};
190
191 std::size_t count{};
192 std::tie(out, count) = prettify_lambda_scope(
193 std::move(out),
194 lambdaEndRegex,
195 StringViewT{scope.cbegin(), scope.cbegin() + libcxxLambdaScopePrefix.size()},
196 scope);
197 return std::tuple{std::move(out), 0u, count};
198 }
199 // while libstdc++ sometimes uses `{lambda()#N}`
200 #else
201 if (constexpr StringViewT lambdaScopePrefix{"{lambda("};
202 scope.starts_with(lambdaScopePrefix))
203 {
204 static RegexT lambdaEndRegex{R"(\)#(\d+)\}::$)"};
205
206 std::size_t count{};
207 std::tie(out, count) = prettify_lambda_scope(
208 std::move(out),
209 lambdaEndRegex,
210 StringViewT{scope.cbegin(), scope.cbegin() + lambdaScopePrefix.size()},
211 scope);
212 return std::tuple{std::move(out), 0u, count};
213 }
214 #endif
215
216 static const RegexT functionSuffix{
217 R"(\))"
218 R"(\s*(?:const)?)"
219 R"(\s*(?:volatile)?)"
220 R"(\s*&{0,2})"
221 R"(\s*::$)"};
222 if (std::regex_search(scope.cbegin(), scope.cend(), matches, functionSuffix))
223 {
224 static RegexT const functionScopePrefix{
225 R"(^(?:\w+\s+)?)" // return type (optional)
226 R"((operator.+?|\w+)\‍()" // function-name + (
227 };
228 StringViewT const rest{scope.cbegin(), matches[0].first};
229 SVMatchT prefixMatches{};
230 std::regex_search(rest.cbegin(), rest.cend(), prefixMatches, functionScopePrefix);
231 MIMICPP_ASSERT(!prefixMatches.empty(), "No corresponding function prefix found.");
232
233 std::size_t count{};
234 std::tie(out, count) = prettify_function_scope(
235 std::move(out),
236 prefixMatches,
237 matches);
238 return std::tuple{
239 std::move(out),
240 std::ranges::distance(rest.cbegin(), prefixMatches[0].first),
241 count};
242 }
243
244 // Probably a regular c++-scope
245 return std::tuple{
246 std::ranges::copy(scope, std::move(out)).out,
247 0u,
248 scope.size()};
249 }

255 name = std::regex_replace(name, unifyClosingAngleBrackets, ">");

267 "std::__1::"

281 template <print_iterator OutIter>
282 std::tuple<OutIter, std::size_t, std::size_t> consume_next_scope(OutIter out, StringViewT const scope)
283 {
284 SVMatchT matches{};
285
286 // the anonymous namespace was already prettified, so just print it as-is
287 if (scope.ends_with(anonymousNamespaceTargetScopeText))
288 {
289 return std::tuple{
290 std::ranges::copy(anonymousNamespaceTargetScopeText, std::move(out)).out,
291 scope.size() - anonymousNamespaceTargetScopeText.size(),
292 anonymousNamespaceTargetScopeText.size()};
293 }
294
295 if (constexpr StringViewT templateScopeSuffix{">::"};
296 scope.ends_with(templateScopeSuffix))
297 {
298 return std::tuple{
299 prettify_template_scope(std::move(out), scope, templateScopeSuffix),
300 0u,
301 scope.size()};
302 }
303
304 // was already prettified, thus return as-is
305 if (static RegexT const lambdaScope{R"(lambda#\d+::$)"};
306 std::regex_search(scope.cbegin(), scope.cend(), matches, lambdaScope))
307 {
308 return std::tuple{
309 std::ranges::copy(matches[0].first, matches[0].second, std::move(out)).out,
310 std::ranges::distance(scope.cbegin(), matches[0].first),
311 matches[0].length()};
312 }
313
314 if (static RegexT const regularScope{R"(\w+::$)"};
315 std::regex_search(scope.cbegin(), scope.cend(), matches, regularScope))
316 {
317 return std::tuple{
318 std::ranges::copy(matches[0].first, matches[0].second, std::move(out)).out,
319 std::ranges::distance(scope.cbegin(), matches[0].first),
320 matches[0].length()};
321 }

323 static const RegexT functionSuffix{
324 R"(\))"
325 R"(\s*(?:const)?)"
326 R"(\s*(?:volatile)?)"
327 R"(\s*&{0,2})"
328 R"(\s*'::$)"};
329 if (std::regex_search(scope.cbegin(), scope.cend(), matches, functionSuffix))
330 {
331 static RegexT const functionScopePrefix{
332 "`"
333 R"((?:\w+\s+)?)" // return type (optional)
334 R"((operator.+?|\w+)\‍()" // function-name + (
335 };
336 StringViewT const rest{scope.cbegin(), matches[0].first};
337 SVMatchT prefixMatches{};
338 std::regex_search(rest.cbegin(), rest.cend(), prefixMatches, functionScopePrefix);
339 MIMICPP_ASSERT(!prefixMatches.empty(), "No corresponding function prefix found.");
340
341 std::size_t count{};
342 std::tie(out, count) = prettify_function_scope(
343 std::move(out),
344 prefixMatches,
345 matches);
346 return std::tuple{
347 std::move(out),
348 std::ranges::distance(rest.cbegin(), prefixMatches[0].first),
349 count};
350 }
351
352 // unknown scope. Ideally, this should never be used, but we leave it here as a fallback.
353 // Output will probably quite odd.
354 return std::tuple{
355 std::ranges::copy(scope, std::move(out)).out,
356 0u,
357 scope.size()};
358 }

360 [[nodiscard]]
361 inline StringT apply_general_prettification(StringT name)
362 {
363 static RegexT const omitClassStructEnum{R"(\b(class|struct|enum)\s+)"};
364 name = std::regex_replace(name, omitClassStructEnum, "");
365
366 static RegexT const omitAccessSpecifiers{R"(\b(?:public|private|protected):\s+)"};
367 name = std::regex_replace(name, omitAccessSpecifiers, "");
368
369 static const RegexT terseAnonymousNamespace{"`anonymous namespace'::"};
370 name = std::regex_replace(name, terseAnonymousNamespace, anonymousNamespaceTargetScopeText.data());
371
372 static const RegexT omitVirtualNamespace{R"(`\d+'::)"};
373 name = std::regex_replace(name, omitVirtualNamespace, "");
374
375 static const RegexT prettifyLambda{R"(<lambda_(\d+)>)"};
376 name = std::regex_replace(name, prettifyLambda, "lambda#$1");
377
378 #if MIMICPP_DETAIL_IS_CLANG_CL
379 static RegexT const omitAutoTokens{R"(<auto>\s*)"};
380 name = std::regex_replace(name, omitAutoTokens, "");
381 #endif
382
383 static RegexT const omitStaticSpecifier{R"(\bstatic\s+)"};
384 name = std::regex_replace(name, omitStaticSpecifier, "");
385
386 static RegexT const unifyComma{R"(\s*,\s*)"};
387 name = std::regex_replace(name, unifyComma, ", ");
388
389 static RegexT const unifyClosingAngles{R"(\s*>)"};
390 name = std::regex_replace(name, unifyClosingAngles, ">");
391
392 // something like call-convention and __ptr64
393 static RegexT const omitImplementationSpecifiers{R"(\b__\w+\b\s*)"};
394 name = std::regex_replace(name, omitImplementationSpecifiers, "");
395
396 return name;
397 }

405 constexpr StringViewT argumentDelimiter{","};
406 constexpr std::array openingBrackets{'<', '(', '[', '{'};

409 template <typename OutIter>
410 constexpr OutIter prettify_scopes(OutIter out, StringT name)
411 {
412 while (auto const match = util::find_next_unwrapped_token(
413 name,
414 scopeDelimiter,
415 openingBrackets,
416 closingBrackets))
417 {
418 std::size_t index{};
419 std::size_t count{};
420 std::tie(out, index, count) = consume_next_scope(
421 out,
422 StringViewT{name.cbegin(), match.end()});
423 name.erase(index, count);
424 }
425
426 return out;
427 }

430 inline StringViewT trimmed(StringViewT const str)
431 {
432 constexpr auto is_space = [](CharT const c) {
433 return static_cast<bool>(std::isspace(static_cast<int>(c)));
434 };
435
436 return {
437 std::ranges::find_if_not(str, is_space),
438 std::ranges::find_if_not(str | std::views::reverse, is_space).base()};
439 }

441 [[nodiscard]]
442 inline std::tuple<StringT, std::optional<std::tuple<StringT, StringT, StringT>>> detect_special_type_info(StringT name)
443 {
444 // it's a template type
445 if (name.back() == '>')
446 {
447 auto reversedName = name
448 | std::views::reverse
449 | std::views::drop(1);
450 auto const iter = util::find_closing_token(reversedName, '>', '<');
451 MIMICPP_ASSERT(iter != reversedName.end(), "No template begin found.");
452 std::tuple info{
453 "<",
454 StringT{iter.base(), name.end() - 1},
455 ">"
456 };
457 name.erase(iter.base() - 1, name.cend());
458
459 return {
460 std::move(name),
461 std::move(info)};
462 }
463
464 #if MIMICPP_DETAIL_IS_GCC
465
466 static RegexT lambdaSuffixRegex{R"(#(\d+)}$)"};
467 // it's a lambda
468 if (SMatchT matches{};
469 std::regex_search(name, matches, lambdaSuffixRegex))
470 {
471 auto reversedName = name
472 | std::views::reverse
473 | std::views::drop(matches[0].length());
474 auto const iter = util::find_closing_token(reversedName, '}', '{');
475 MIMICPP_ASSERT(iter != reversedName.end(), "No lambda begin found.");
476
477 std::tuple info{
478 format::format("lambda#{}", StringViewT{matches[1].first, matches[1].second}),
479 "",
480 ""};
481 name.erase(iter.base() - 1, name.cend());
482
483 return {
484 std::move(name),
485 std::move(info)};
486 }

490 static const RegexT functionSuffix{
491 R"(\))"
492 R"(\s*(?:const)?)"
493 R"(\s*(?:volatile)?)"
494 R"(\s*&{0,2})"
495 R"(\s*$)"};
496 if (SMatchT matches{};
497 std::regex_search(name, matches, functionSuffix))
498 {
499 auto reversedName = name
500 | std::views::reverse
501 | std::views::drop(matches[0].length());
502 auto const iter = util::find_closing_token(reversedName, ')', '(');
503 MIMICPP_ASSERT(iter != reversedName.end(), "No function begin found.");
504
505 StringViewT const argList{iter.base(), name.end() - matches[0].length()};
506 std::tuple info{
507 "(",

509 StringT{matches[0].first, matches[0].second}

512
513 return {
514 std::move(name),
515 std::move(info)};

519 std::move(name),
520 std::nullopt};

526 template <print_iterator OutIter>
527 constexpr OutIter prettify_identifier(OutIter out, StringT name)
528 {
529 name = detail::apply_general_prettification(std::move(name));
530
531 std::optional<std::tuple<StringT, StringT, StringT>> specialTypeInfo{};
532 std::tie(name, specialTypeInfo) = detail::detect_special_type_info(std::move(name));
533
534 if (auto reversedName = name | std::views::reverse;
535 auto const lastScopeMatch = util::find_next_unwrapped_token(
536 reversedName,
537 detail::scopeDelimiter,
538 detail::closingBrackets,
539 detail::openingBrackets))
540 {
541 std::size_t const splitIndex = std::ranges::distance(name.cbegin(), lastScopeMatch.begin().base());
542 StringT topLevelIdentifier = name.substr(splitIndex);
543
544 // includes last ::
545 name.erase(splitIndex);
546 out = detail::prettify_scopes(std::move(out), std::move(name));
547
548 name = std::move(topLevelIdentifier);
549 }
550
551 out = std::ranges::copy(name, std::move(out)).out;
552
553 if (specialTypeInfo)
554 {
555 auto const& [scopeBegin, argList, scopeEnd] = *specialTypeInfo;
556 out = std::ranges::copy(scopeBegin, std::move(out)).out;
557
558 bool isFirst{true};
559 StringViewT pendingArgList{argList};
560 while (!pendingArgList.empty())
561 {
562 if (!std::exchange(isFirst, false))
563 {
564 out = std::ranges::copy(StringViewT{", "}, std::move(out)).out;
565 }
566
567 auto const tokenDelimiter = util::find_next_unwrapped_token(
568 pendingArgList,
569 detail::argumentDelimiter,
570 detail::openingBrackets,
571 detail::closingBrackets);
572 StringViewT const arg = detail::trimmed(StringViewT{pendingArgList.begin(), tokenDelimiter.begin()});
573 out = type::prettify_identifier(std::move(out), StringT{arg});
574 pendingArgList = StringViewT{tokenDelimiter.end(), pendingArgList.end()};
575 }
576
577 out = std::ranges::copy(scopeEnd, std::move(out)).out;
578 }
579
580 return out;
581 }
