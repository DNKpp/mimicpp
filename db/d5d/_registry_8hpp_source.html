116 template <typename Signature>
117 [[nodiscard]]
118 signature_return_type_t<Signature> handle_call(reporting::TargetReport target, call::info_for_signature_t<Signature> call)
119 {
120 std::scoped_lock const lock{m_ExpectationsMx};
121 auto&& [matches, inapplicableMatches, noMatches] = evaluate_expectations(target, call);
122
123 std::size_t const stacktraceSkip{1u + call.baseStacktraceSkip};
124 if (!std::ranges::empty(matches))
125 {
126 return handle_matched_call<Signature>(std::move(target), std::move(call), matches, stacktraceSkip + 1u);
127 }
128
129 if (!std::ranges::empty(inapplicableMatches))
130 {
131 auto reports = inapplicableMatches | std::views::transform([](auto const& exp) { return exp->report(); });
132 reporting::detail::report_inapplicable_matches(
133 reporting::make_call_report(std::move(target), std::move(call), util::stacktrace::current(stacktraceSkip)),
134 std::vector(reports.begin(), reports.end()));
135 }
136
137 reporting::detail::report_no_matches(
138 reporting::make_call_report(std::move(target), std::move(call), util::stacktrace::current(stacktraceSkip)),
139 make_no_match_reports(std::move(noMatches)));
140 }

146 [[nodiscard]]
147 static std::optional<reporting::RequirementOutcomes> try_match_call(
148 reporting::TargetReport const& target,
149 auto const& callInfo,
150 Expectation const& expectation) noexcept
151 {
152 try
153 {
154 return expectation.matches(callInfo);
155 }
156 catch (...)
157 {
158 reporting::detail::report_unhandled_exception(
159 reporting::make_call_report(
160 target,
161 callInfo,
162 util::stacktrace::current(3u + callInfo.baseStacktraceSkip)),
163 expectation.report(),
164 std::current_exception());
165 }
166
167 return std::nullopt;
168 }

170 [[nodiscard]]
171 auto evaluate_expectations(reporting::TargetReport const& target, auto const& callInfo)
172 {
173 std::vector<Expectation*> matches{};
174 std::vector<Expectation const*> inapplicableMatches{};
175 std::vector<std::tuple<Expectation const*, reporting::RequirementOutcomes>> noMatches{};
176
177 for (auto& exp : std::views::reverse(m_Expectations))
178 {
179 if (std::optional result = try_match_call(target, callInfo, exp))
180 {
181 if (std::ranges::any_of(result->outcomes, [](auto const& el) { return el == false; }))
182 {
183 noMatches.emplace_back(&exp, *std::move(result));
184 }
185 else if (!exp.is_applicable())
186 {
187 inapplicableMatches.emplace_back(&exp);
188 }
189 else
190 {
191 matches.emplace_back(&exp);
192 }
193 }
194 }
195
196 return std::make_tuple(std::move(matches), std::move(inapplicableMatches), std::move(noMatches));
197 }

199 template <typename Signature>
200 [[nodiscard]]
201 signature_return_type_t<Signature> handle_matched_call(
202 reporting::TargetReport&& target,
203 call::info_for_signature_t<Signature>&& callInfo,
204 std::span<Expectation* const> const matches,
205 std::size_t const stacktraceSkip)
206 {
207 std::vector reports = std::invoke([&] {
208 auto view = std::views::transform(matches, [](auto const& exp) { return exp->report(); });
209 return std::vector(view.begin(), view.end());
210 });
211 MIMICPP_ASSERT(matches.size() == reports.size(), "Size mismatch.");
212 auto const bestMatchIter = std::ranges::max_element(
213 reports,
214 std::not_fn(&sequence::detail::has_better_rating),
215 [](auto const& el) noexcept -> const auto& {
216 return std::get<reporting::state_applicable>(el.controlReport).sequenceRatings;
217 });
218 auto& expectation = *matches[std::ranges::distance(reports.cbegin(), bestMatchIter)];
219
220 if (settings::report_success())
221 {
222 // Todo: Avoid the call copy
223 // Maybe we can prevent the copy here, but we should keep the instruction order as-is,
224 // because in cases of a throwing finalizer, we might introduce bugs.
225 // At least there are some tests, which will fail if done incorrectly.
226 reporting::detail::report_full_match(
227 reporting::make_call_report(std::move(target), callInfo, util::stacktrace::current(stacktraceSkip)),
228 std::move(*bestMatchIter));
229 }
230
231 return expectation.consume(callInfo);
232 }

235 static std::vector<reporting::NoMatchReport> make_no_match_reports(std::vector<std::tuple<Expectation const*, reporting::RequirementOutcomes>>&& outcomes)
236 {
237 std::vector<reporting::NoMatchReport> reports{};
238 reports.reserve(outcomes.size());
239 for (auto&& [expectationPtr, outcome] : std::move(outcomes))
240 {
241 reports.emplace_back(expectationPtr->report(), std::move(outcome));
242 }
